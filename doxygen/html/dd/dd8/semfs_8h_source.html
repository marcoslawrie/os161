47 /*
48  * A user-facing semaphore.
49  *
50  * We don't use the kernel-level semaphore to implement it (although
51  * that would be tidy) because we'd have to violate its abstraction.
52  * XXX: or would we? review once all this is done.
53  */
54 struct semfs_sem {
55  struct lock *sems_lock; /* Lock to protect count */
56  struct cv *sems_cv; /* CV to wait */
57  unsigned sems_count; /* Semaphore count */
58  bool sems_hasvnode; /* The vnode exists */
59  bool sems_linked; /* In the directory */
60 };
61 DECLARRAY(semfs_sem, SEMFS_INLINE);

72 /*
73  * Vnode. These are separate from the semaphore structures so they can
74  * come and go at the whim of VOP_RECLAIM. (It might seem tidier to
75  * ignore VOP_RECLAIM and destroy vnodes only when the underlying
76  * objects are removed; but it ends up being more complicated in
77  * practice. XXX: review after finishing)
78  */
79 struct semfs_vnode {
80  struct vnode semv_absvn; /* Abstract vnode */
81  struct semfs *semv_semfs; /* Back-pointer to fs */
82  unsigned semv_semnum; /* Which semaphore */
83 };

85 /*

98 };
