124 /*
125  * Find and create secondary CPUs.
126  */
127 void
128 lamebus_find_cpus(struct lamebus_softc *lamebus)
129 {
130  uint32_t mainboard_vid, mainboard_did;
131  uint32_t cpumask, self, bit, val;
132  unsigned i, numcpus, bootcpu;
133  unsigned hwnum[32];
134 
135  mainboard_vid = read_cfg_register(lamebus, LB_CONTROLLER_SLOT,
136  CFGREG_VID);
137  mainboard_did = read_cfg_register(lamebus, LB_CONTROLLER_SLOT,
138  CFGREG_DID);
139  if (mainboard_vid == LB_VENDOR_CS161 &&
140  mainboard_did == LBCS161_UPBUSCTL) {
141  /* Old uniprocessor mainboard; no cpu registers. */
142  lamebus->ls_uniprocessor = 1;
143  return;
144  }
145 
146  cpumask = read_ctl_register(lamebus, CTLREG_CPUS);
147  self = read_ctl_register(lamebus, CTLREG_SELF);
148 
149  numcpus = 0;
150  bootcpu = 0;
151  for (i=0; i<32; i++) {
152  bit = (uint32_t)1 << i;
153  if ((cpumask & bit) != 0) {
154  if (self & bit) {
155  bootcpu = numcpus;
156  curcpu->c_hardware_number = i;
157  }
158  hwnum[numcpus] = i;
159  numcpus++;
160  }
161  }
162 
163  for (i=0; i<numcpus; i++) {
164  if (i != bootcpu) {
165  cpu_create(hwnum[i]);
166  }
167  }
168 
169  /*
170  * By default, route all interrupts only to the boot cpu. We
171  * could be arbitrarily more elaborate, up to things like
172  * dynamic load balancing.
173  */
174 
175  for (i=0; i<numcpus; i++) {
176  if (i != bootcpu) {
177  val = 0;
178  }
179  else {
180  val = 0xffffffff;
181  }
182  write_ctlcpu_register(lamebus, hwnum[i], CTLCPU_CIRQE, val);
183  }
184 }

186 /*
187  * Start up secondary CPUs.
188  *
189  * The first word of the CRAM area is set to the entry point for new
190  * CPUs; the second to the (software) CPU number. Note that the logic
191  * here assumes the boot CPU is CPU 0 and the others are 1-N as
192  * created in the function above. This is fine if all CPUs are on
193  * LAMEbus; if in some environment there are other CPUs about as well
194  * this logic will have to be made more complex.
195  */
196 void
197 lamebus_start_cpus(struct lamebus_softc *lamebus)
198 {
199  uint32_t cpumask, self, bit;
200  uint32_t ctlcpuoffset;
201  uint32_t *cram;
202  unsigned i;
203  unsigned cpunum;
204 
205  if (lamebus->ls_uniprocessor) {
206  return;
207  }
208 
209  cpumask = read_ctl_register(lamebus, CTLREG_CPUS);
210  self = read_ctl_register(lamebus, CTLREG_SELF);
211 
212  /* Poke in the startup address. */
213  cpunum = 1;
214  for (i=0; i<32; i++) {
215  bit = (uint32_t)1 << i;
216  if ((cpumask & bit) != 0) {
217  if (self & bit) {
218  continue;
219  }
220  ctlcpuoffset = LB_CTLCPU_OFFSET + i * LB_CTLCPU_SIZE;
221  cram = lamebus_map_area(lamebus,
222  LB_CONTROLLER_SLOT,
223  ctlcpuoffset + CTLCPU_CRAM);
224  cram[0] = (uint32_t)cpu_start_secondary;
225  cram[1] = cpunum++;
226  }
227  }
228  /* Ensure all the above writes get flushed. */
229  membar_store_store();
230 
231  /* Now, enable them all. */
232  write_ctl_register(lamebus, CTLREG_CPUE, cpumask);
233 }

235 /*
236  * Probe function.
237  *
238  * Given a LAMEbus, look for a device that's not already been marked
239  * in use, has the specified IDs, and has a device revision level at
240  * least as high as the minimum specified.
241  *
242  * Returns the slot number found (0-31) or -1 if nothing suitable was
243  * found.
244  *
245  * If VERSION_RET is not null, return the device version found. This
246  * allows drivers to blacklist specific versions or otherwise conduct
247  * more specific checks.
248  */
249 
250 int
251 lamebus_probe(struct lamebus_softc *sc,
252  uint32_t vendorid, uint32_t deviceid,
253  uint32_t lowver, uint32_t *version_ret)
254 {
255  int slot;
256  uint32_t val;
257 
258  /*
259  * Because the slot information in sc is used when dispatching
260  * interrupts, disable interrupts while working with it.
261  */
262 
263  spinlock_acquire(&sc->ls_lock);
264 
265  for (slot=0; slot<LB_NSLOTS; slot++) {
266  if (sc->ls_slotsinuse & (1<<slot)) {
267  /* Slot already in use; skip */
268  continue;
269  }
270 
271  val = read_cfg_register(sc, slot, CFGREG_VID);
272  if (val!=vendorid) {
273  /* Wrong vendor id */
274  continue;
275  }
276 
277  val = read_cfg_register(sc, slot, CFGREG_DID);
278  if (val != deviceid) {
279  /* Wrong device id */
280  continue;
281  }
282 
283  val = read_cfg_register(sc, slot, CFGREG_DRL);
284  if (val < lowver) {
285  /* Unsupported device revision */
286  continue;
287  }
288  if (version_ret != NULL) {
289  *version_ret = val;
290  }
291 
292  /* Found something */
293 
294  spinlock_release(&sc->ls_lock);
295  return slot;
296  }
297 
298  /* Found nothing */
299 
300  spinlock_release(&sc->ls_lock);
301  return -1;
302 }

304 /*
305  * Mark that a slot is in use.
306  * This prevents the probe routine from returning the same device over
307  * and over again.
308  */
309 void
310 lamebus_mark(struct lamebus_softc *sc, int slot)
311 {
312  uint32_t mask = ((uint32_t)1) << slot;
313  KASSERT(slot>=0 && slot < LB_NSLOTS);
314 
315  spinlock_acquire(&sc->ls_lock);
316 
317  if ((sc->ls_slotsinuse & mask)!=0) {
318  panic("lamebus_mark: slot %d already in use\n", slot);
319  }
320 
321  sc->ls_slotsinuse |= mask;
322 
323  spinlock_release(&sc->ls_lock);
324 }

429 /*
430  * LAMEbus interrupt handling function. (Machine-independent!)
431  */
432 void
433 lamebus_interrupt(struct lamebus_softc *lamebus)
434 {
435  /*
436  * Note that despite the fact that "spl" stands for "set
437  * priority level", we don't actually support interrupt
438  * priorities. When an interrupt happens, we look through the
439  * slots to find the first interrupting device and call its
440  * interrupt routine, no matter what that device is.
441  *
442  * Note that the entire LAMEbus uses only one on-cpu interrupt line.
443  * Thus, we do not use any on-cpu interrupt priority system either.
444  */
445 
446  int slot;
447  uint32_t mask;
448  uint32_t irqs;
449  void (*handler)(void *);
450  void *data;
451 
452  /* For keeping track of how many bogus things happen in a row. */
453  static int duds = 0;
454  int duds_this_time = 0;
455 
456  /* and we better have a valid bus instance. */
457  KASSERT(lamebus != NULL);
458 
459  /* Lock the softc */
460  spinlock_acquire(&lamebus->ls_lock);
461 
462  /*
463  * Read the LAMEbus controller register that tells us which
464  * slots are asserting an interrupt condition.
465  */
466  irqs = read_ctl_register(lamebus, CTLREG_IRQS);
467 
468  if (irqs == 0) {
469  /*
470  * Huh? None of them? Must be a glitch.
471  */
472  kprintf("lamebus: stray interrupt on cpu %u\n",
473  curcpu->c_number);
474  duds++;
475  duds_this_time++;
476 
477  /*
478  * We could just return now, but instead we'll
479  * continue ahead. Because irqs == 0, nothing in the
480  * loop will execute, and passing through it gets us
481  * to the code that checks how many duds we've
482  * seen. This is important, because we just might get
483  * a stray interrupt that latches itself on. If that
484  * happens, we're pretty much toast, but it's better
485  * to panic and hopefully reset the system than to
486  * loop forever printing "stray interrupt".
487  */
488  }
489 
490  /*
491  * Go through the bits in the value we got back to see which
492  * ones are set.
493  */
494 
495  for (mask=1, slot=0; slot<LB_NSLOTS; mask<<=1, slot++) {
496  if ((irqs & mask) == 0) {
497  /* Nope. */
498  continue;
499  }
500 
501  /*
502  * This slot is signalling an interrupt.
503  */
504 
505  if ((lamebus->ls_slotsinuse & mask)==0) {
506  /*
507  * No device driver is using this slot.
508  */
509  duds++;
510  duds_this_time++;
511  continue;
512  }
513 
514  if (lamebus->ls_irqfuncs[slot]==NULL) {
515  /*
516  * The device driver hasn't installed an interrupt
517  * handler.
518  */
519  duds++;
520  duds_this_time++;
521  continue;
522  }
523 
524  /*
525  * Call the interrupt handler. Release the spinlock
526  * while we do so, in case other CPUs are handling
527  * interrupts on other devices.
528  */
529  handler = lamebus->ls_irqfuncs[slot];
530  data = lamebus->ls_devdata[slot];
531  spinlock_release(&lamebus->ls_lock);
532 
533  handler(data);
534 
535  spinlock_acquire(&lamebus->ls_lock);
536 
537  /*
538  * Reload the mask of pending IRQs - if we just called
539  * hardclock, we might not have come back to this
540  * context for some time, and it might have changed.
541  */
542 
543  irqs = read_ctl_register(lamebus, CTLREG_IRQS);
544  }
545 
546 
547  /*
548  * If we get interrupts for a slot with no driver or no
549  * interrupt handler, it's fairly serious. Because LAMEbus
550  * uses level-triggered interrupts, if we don't shut off the
551  * condition, we'll keep getting interrupted continuously and
552  * the system will make no progress. But we don't know how to
553  * do that if there's no driver or no interrupt handler.
554  *
555  * So, if we get too many dud interrupts, panic, since it's
556  * better to panic and reset than to hang.
557  *
558  * If we get through here without seeing any duds this time,
559  * the condition, whatever it was, has gone away. It might be
560  * some stupid device we don't have a driver for, or it might
561  * have been an electrical transient. In any case, warn and
562  * clear the dud count.
563  */
564 
565  if (duds_this_time == 0 && duds > 0) {
566  kprintf("lamebus: %d dud interrupts\n", duds);
567  duds = 0;
568  }
569 
570  if (duds > 10000) {
571  panic("lamebus: too many (%d) dud interrupts\n", duds);
572  }
573 
574  /* Unlock the softc */
575  spinlock_release(&lamebus->ls_lock);
576 }

640 {
641  struct lamebus_softc *lamebus;
642  int i;
643 
644  /* Allocate space for lamebus data */
645  lamebus = kmalloc(sizeof(struct lamebus_softc));
646  if (lamebus==NULL) {
647  panic("lamebus_init: Out of memory\n");
648  }
649 
650  spinlock_init(&lamebus->ls_lock);
651 
652  /*
653  * Initialize the LAMEbus data structure.
654  */
655  lamebus->ls_slotsinuse = 1 << LB_CONTROLLER_SLOT;
656 
657  for (i=0; i<LB_NSLOTS; i++) {
658  lamebus->ls_devdata[i] = NULL;
659  lamebus->ls_irqfuncs[i] = NULL;
660  }
661 
662  lamebus->ls_uniprocessor = 0;
663 
664  return lamebus;
