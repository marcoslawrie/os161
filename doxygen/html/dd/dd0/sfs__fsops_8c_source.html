53 /*
54  * Routine for doing I/O (reads or writes) on the free block bitmap.
55  * We always do the whole bitmap at once; writing individual sectors
56  * might or might not be a worthwhile optimization.
57  *
58  * The free block bitmap consists of SFS_FREEMAPBLOCKS 512-byte
59  * sectors of bits, one bit for each sector on the filesystem. The
60  * number of blocks in the bitmap is thus rounded up to the nearest
61  * multiple of 512*8 = 4096. (This rounded number is SFS_FREEMAPBITS.)
62  * This means that the bitmap will (in general) contain space for some
63  * number of invalid sectors that are actually beyond the end of the
64  * disk device. This is ok. These sectors are supposed to be marked
65  * "in use" by mksfs and never get marked "free".
66  *
67  * The sectors used by the superblock and the bitmap itself are
68  * likewise marked in use by mksfs.
69  */
70 static
71 int
72 sfs_freemapio(struct sfs_fs *sfs, enum uio_rw rw)
73 {
74  uint32_t j, freemapblocks;
75  char *freemapdata;
76  int result;
77 
78  /* Number of blocks in the free block bitmap. */
79  freemapblocks = SFS_FS_FREEMAPBLOCKS(sfs);
80 
81  /* Pointer to our freemap data in memory. */
82  freemapdata = bitmap_getdata(sfs->sfs_freemap);
83 
84  /* For each block in the free block bitmap... */
85  for (j=0; j<freemapblocks; j++) {
86 
87  /* Get a pointer to its data */
88  void *ptr = freemapdata + j*SFS_BLOCKSIZE;
89 
90  /* and read or write it. The freemap starts at sector 2. */
91  if (rw == UIO_READ) {
92  result = sfs_readblock(sfs, SFS_FREEMAP_START+j, ptr,
93  SFS_BLOCKSIZE);
94  }
95  else {
96  result = sfs_writeblock(sfs, SFS_FREEMAP_START+j, ptr,
97  SFS_BLOCKSIZE);
98  }
99 
100  /* If we failed, stop. */
101  if (result) {
102  return result;
103  }
104  }
105  return 0;
106 }

166 /*
167  * Sync routine. This is what gets invoked if you do FS_SYNC on the
168  * sfs filesystem structure.
169  */
170 static
171 int
172 sfs_sync(struct fs *fs)
173 {
174  struct sfs_fs *sfs;
175  int result;
176 
177  vfs_biglock_acquire();
178 
179  /*
180  * Get the sfs_fs from the generic abstract fs.
181  *
182  * Note that the abstract struct fs, which is all the VFS
183  * layer knows about, is actually a member of struct sfs_fs.
184  * The pointer in the struct fs points back to the top of the
185  * struct sfs_fs - essentially the same object. This can be a
186  * little confusing at first.
187  *
188  * The following diagram may help:
189  *
190  * struct sfs_fs <-------------\
191  * : |
192  * : sfs_absfs (struct fs) | <------\
193  * : : | |
194  * : : various members | |
195  * : : | |
196  * : : fs_data ----------/ |
197  * : : ...|...
198  * : . VFS .
199  * : . layer .
200  * : other members .......
201  * :
202  * :
203  *
204  * This construct is repeated with vnodes and devices and other
205  * similar things all over the place in OS/161, so taking the
206  * time to straighten it out in your mind is worthwhile.
207  */
208 
209  sfs = fs->fs_data;
210 
211  /* If any vnodes need to be written, write them. */
212  result = sfs_sync_vnodes(sfs);
213  if (result) {
214  vfs_biglock_release();
215  return result;
216  }
217 
218  /* If the free block map needs to be written, write it. */
219  result = sfs_sync_freemap(sfs);
220  if (result) {
221  vfs_biglock_release();
222  return result;
223  }
224 
225  /* If the superblock needs to be written, write it. */
226  result = sfs_sync_superblock(sfs);
227  if (result) {
228  vfs_biglock_release();
229  return result;
230  }
231 
232  vfs_biglock_release();
233  return 0;
234 }

270 /*
271  * Unmount code.
272  *
273  * VFS calls FS_SYNC on the filesystem prior to unmounting it.
274  */
275 static
276 int
277 sfs_unmount(struct fs *fs)
278 {
279  struct sfs_fs *sfs = fs->fs_data;
280 
281  vfs_biglock_acquire();
282 
283  /* Do we have any files open? If so, can't unmount. */
284  if (vnodearray_num(sfs->sfs_vnodes) > 0) {
285  vfs_biglock_release();
286  return EBUSY;
287  }
288 
289  /* We should have just had sfs_sync called. */
290  KASSERT(sfs->sfs_superdirty == false);
291  KASSERT(sfs->sfs_freemapdirty == false);
292 
293  /* The vfs layer takes care of the device for us */
294  sfs->sfs_device = NULL;
295 
296  /* Destroy the fs object; once we start nuking stuff we can't fail. */
297  sfs_fs_destroy(sfs);
298 
299  /* nothing else to do */
300  vfs_biglock_release();
301  return 0;
302 }

314 /*
315  * Basic constructor for struct sfs_fs. This initializes all fields
316  * but skips stuff that requires reading the volume, like allocating
317  * the freemap.
318  */
319 static
320 struct sfs_fs *
321 sfs_fs_create(void)
322 {
323  struct sfs_fs *sfs;
324 
325  /*
326  * Make sure our on-disk structures aren't messed up
327  */
328  COMPILE_ASSERT(sizeof(struct sfs_superblock)==SFS_BLOCKSIZE);
329  COMPILE_ASSERT(sizeof(struct sfs_dinode)==SFS_BLOCKSIZE);
330  COMPILE_ASSERT(SFS_BLOCKSIZE % sizeof(struct sfs_direntry) == 0);
331 
332  /* Allocate object */
333  sfs = kmalloc(sizeof(struct sfs_fs));
334  if (sfs==NULL) {
335  goto fail;
336  }
337 
338  /*
339  * Fill in fields
340  */
341 
342  /* abstract vfs-level fs */
343  sfs->sfs_absfs.fs_data = sfs;
344  sfs->sfs_absfs.fs_ops = &sfs_fsops;
345 
346  /* superblock */
347  /* (ignore sfs_super, we'll read in over it shortly) */
348  sfs->sfs_superdirty = false;
349 
350  /* device we mount on */
351  sfs->sfs_device = NULL;
352 
353  /* vnode table */
354  sfs->sfs_vnodes = vnodearray_create();
355  if (sfs->sfs_vnodes == NULL) {
356  goto cleanup_object;
357  }
358 
359  /* freemap */
360  sfs->sfs_freemap = NULL;
361  sfs->sfs_freemapdirty = false;
362 
363  return sfs;
364 
365 cleanup_object:
366  kfree(sfs);
367 fail:
368  return NULL;
369 }

371 /*
372  * Mount routine.
373  *
374  * The way mount works is that you call vfs_mount and pass it a
375  * filesystem-specific mount routine. Said routine takes a device and
376  * hands back a pointer to an abstract filesystem. You can also pass
377  * a void pointer through.
378  *
379  * This organization makes cleanup on error easier. Hint: it may also
380  * be easier to synchronize correctly; it is important not to get two
381  * filesystems with the same name mounted at once, or two filesystems
382  * mounted on the same device at once.
383  */
384 static
385 int
386 sfs_domount(void *options, struct device *dev, struct fs **ret)
387 {
388  int result;
389  struct sfs_fs *sfs;
390 
391  vfs_biglock_acquire();
392 
393  /* We don't pass any options through mount */
394  (void)options;
395 
396  /*
397  * We can't mount on devices with the wrong sector size.
398  *
399  * (Note: for all intents and purposes here, "sector" and
400  * "block" are interchangeable terms. Technically a filesystem
401  * block may be composed of several hardware sectors, but we
402  * don't do that in sfs.)
403  */
404  if (dev->d_blocksize != SFS_BLOCKSIZE) {
405  vfs_biglock_release();
406  kprintf("sfs: Cannot mount on device with blocksize %zu\n",
407  dev->d_blocksize);
408  return ENXIO;
409  }
410 
411  sfs = sfs_fs_create();
412  if (sfs == NULL) {
413  vfs_biglock_release();
414  return ENOMEM;
415  }
416 
417  /* Set the device so we can use sfs_readblock() */
418  sfs->sfs_device = dev;
419 
420  /* Load superblock */
421  result = sfs_readblock(sfs, SFS_SUPER_BLOCK, &sfs->sfs_sb,
422  sizeof(sfs->sfs_sb));
423  if (result) {
424  sfs->sfs_device = NULL;
425  sfs_fs_destroy(sfs);
426  vfs_biglock_release();
427  return result;
428  }
429 
430  /* Make some simple sanity checks */
431 
432  if (sfs->sfs_sb.sb_magic != SFS_MAGIC) {
433  kprintf("sfs: Wrong magic number in superblock "
434  "(0x%x, should be 0x%x)\n",
435  sfs->sfs_sb.sb_magic,
436  SFS_MAGIC);
437  sfs->sfs_device = NULL;
438  sfs_fs_destroy(sfs);
439  vfs_biglock_release();
440  return EINVAL;
441  }
442 
443  if (sfs->sfs_sb.sb_nblocks > dev->d_blocks) {
444  kprintf("sfs: warning - fs has %u blocks, device has %u\n",
445  sfs->sfs_sb.sb_nblocks, dev->d_blocks);
446  }
447 
448  /* Ensure null termination of the volume name */
449  sfs->sfs_sb.sb_volname[sizeof(sfs->sfs_sb.sb_volname)-1] = 0;
450 
451  /* Load free block bitmap */
452  sfs->sfs_freemap = bitmap_create(SFS_FS_FREEMAPBITS(sfs));
453  if (sfs->sfs_freemap == NULL) {
454  sfs->sfs_device = NULL;
455  sfs_fs_destroy(sfs);
456  vfs_biglock_release();
457  return ENOMEM;
458  }
459  result = sfs_freemapio(sfs, UIO_READ);
460  if (result) {
461  sfs->sfs_device = NULL;
462  sfs_fs_destroy(sfs);
463  vfs_biglock_release();
464  return result;
465  }
466 
467  /* Hand back the abstract fs */
468  *ret = &sfs->sfs_absfs;
469 
470  vfs_biglock_release();
471  return 0;
472 }
