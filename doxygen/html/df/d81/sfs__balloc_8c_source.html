54 /*
55  * Allocate a block.
56  */
57 int
58 sfs_balloc(struct sfs_fs *sfs, daddr_t *diskblock)
59 {
60  int result;
61 
62  result = bitmap_alloc(sfs->sfs_freemap, diskblock);
63  if (result) {
64  return result;
65  }
66  sfs->sfs_freemapdirty = true;
67 
68  if (*diskblock >= sfs->sfs_sb.sb_nblocks) {
69  panic("sfs: %s: balloc: invalid block %u\n",
70  sfs->sfs_sb.sb_volname, *diskblock);
71  }
72 
73  /* Clear block before returning it */
74  result = sfs_clearblock(sfs, *diskblock);
75  if (result) {
76  bitmap_unmark(sfs->sfs_freemap, *diskblock);
77  }
78  return result;
79 }

81 /*
82  * Free a block.
83  */
84 void
85 sfs_bfree(struct sfs_fs *sfs, daddr_t diskblock)
86 {
87  bitmap_unmark(sfs->sfs_freemap, diskblock);
88  sfs->sfs_freemapdirty = true;
89 }

91 /*
92  * Check if a block is in use.
93  */
94 int
95 sfs_bused(struct sfs_fs *sfs, daddr_t diskblock)
96 {
97  if (diskblock >= sfs->sfs_sb.sb_nblocks) {
98  panic("sfs: %s: sfs_bused called on out of range block %u\n",
99  sfs->sfs_sb.sb_volname, diskblock);
100  }
101  return bitmap_isset(sfs->sfs_freemap, diskblock);
102 }
