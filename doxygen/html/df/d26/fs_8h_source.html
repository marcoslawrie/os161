47 /*
48  * Abstraction operations on a file system:
49  *
50  * fsop_sync - Flush all dirty buffers to disk.
51  * fsop_getvolname - Return volume name of filesystem.
52  * fsop_getroot - Return root vnode of filesystem.
53  * fsop_unmount - Attempt unmount of filesystem.
54  *
55  * fsop_getvolname may return NULL on filesystem types that don't
56  * support the concept of a volume name. The string returned is
57  * assumed to point into the filesystem's private storage and live
58  * until unmount time.
59  *
60  * If the volume name changes on the fly, there is no way at present
61  * to make sure such changes don't cause name conflicts. So it probably
62  * should be considered fixed.
63  *
64  * fsop_getroot should increment the refcount of the vnode returned.
65  * It should not ever return NULL.
66  *
67  * If fsop_unmount returns an error, the filesystem stays mounted, and
68  * consequently the struct fs instance should remain valid. On success,
69  * however, the filesystem object and all storage associated with the
70  * filesystem should have been discarded/released.
71  */
72 struct fs_ops {
73  int (*fsop_sync)(struct fs *);
74  const char *(*fsop_getvolname)(struct fs *);
75  int (*fsop_getroot)(struct fs *, struct vnode **);
76  int (*fsop_unmount)(struct fs *);
77 };
