40 /* Lowest revision we support */
41 #define LOW_VERSION 1

43 struct emu_softc *
44 attach_emu_to_lamebus(int emuno, struct lamebus_softc *sc)
45 {
46  struct emu_softc *es;
47  int slot = lamebus_probe(sc, LB_VENDOR_CS161, LBCS161_EMUFS,
48  LOW_VERSION, NULL);
49  if (slot < 0) {
50  return NULL;
51  }
52 
53  es = kmalloc(sizeof(struct emu_softc));
54  if (es==NULL) {
55  return NULL;
56  }
57 
58  es->e_busdata = sc;
59  es->e_buspos = slot;
60  es->e_unit = emuno;
61 
62  lamebus_mark(sc, slot);
63  lamebus_attach_interrupt(sc, slot, es, emu_irq);
64 
65  return es;
66 }
