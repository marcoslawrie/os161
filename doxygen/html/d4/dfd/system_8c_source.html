44 int
45 system(const char *cmd)
46 {
47  /*
48  * Ordinarily, you call the shell to process the command.
49  * But we don't know that the shell can do that. So, do it
50  * ourselves.
51  */
52 
53  char tmp[MAXCMDSIZE];
54  char *argv[MAXARGS+1];
55  int nargs=0;
56  char *s;
57  int pid, status;
58 
59  if (strlen(cmd) >= sizeof(tmp)) {
60  errno = E2BIG;
61  return -1;
62  }
63  strcpy(tmp, cmd);
64 
65  for (s = strtok(tmp, " \t"); s; s = strtok(NULL, " \t")) {
66  if (nargs < MAXARGS) {
67  argv[nargs++] = s;
68  }
69  else {
70  errno = E2BIG;
71  return 1;
72  }
73  }
74 
75  argv[nargs] = NULL;
76 
77  pid = fork();
78  switch (pid) {
79  case -1:
80  return -1;
81  case 0:
82  /* child */
83  execv(argv[0], argv);
84  /* exec only returns if it fails */
85  _exit(255);
86  default:
87  /* parent */
88  waitpid(pid, &status, 0);
89  return status;
90  }
91 }
