31  * crash.c
32  *
33  * Commit a variety of exceptions, primarily address faults.
34  *
35  * Once the basic system calls assignment is complete, none of these
36  * should crash the kernel.
37  *
38  * They should all, however, terminate this program, except for the
39  * one that writes to the code segment. (That one won't cause program
40  * termination until/unless you implement read-only segments in your
41  * VM system.)
42  */

173 void
174 alignment_error(void)
175 {
176  int x;
177  int *ptr, *badptr;
178  volatile uintptr_t ptrval;
179  volatile int j;
180 
181  x = 0;
182  ptr = &x;
183  /*
184  * Try to hide what's going on from gcc; gcc 4.8 seems to
185  * detect the unaligned access and issue unaligned read
186  * instructions for it, so then it doesn't fault. Feh.
187  */
188  ptrval = (uintptr_t)ptr;
189  ptrval++;
190  badptr = (int *)ptrval;
191 
192  j = *badptr;
193 
194  // gcc 4.8 improperly demands this
195  (void)j;
196 }

266 static
267 void
268 runop(int op)
269 {
270  int opindex;
271  pid_t pid;
272  int status;
273  int ok;
274 
275  if (op=='*') {
276  for (unsigned i=0; ops[i].name; i++) {
277  runop(ops[i].ch);
278  }
279  return;
280  }
281  else if (op == '-') {
282  forking = 0;
283  warnx("Forking disabled - next try will be the last");
284  return;
285  }
286  else if (op == '+') {
287  forking = 1;
288  warnx("Forking enabled.");
289  return;
290  }
291 
292  /* intentionally don't check if op is in bounds :) */
293  opindex = op-'a';
294 
295  printf("Running: [%c] %s\n", ops[opindex].ch, ops[opindex].name);
296 
297  if (forking) {
298  pid = fork();
299  if (pid < 0) {
300  /* error */
301  err(1, "fork");
302  }
303  else if (pid > 0) {
304  /* parent */
305  if (waitpid(pid, &status, 0) < 0) {
306  err(1, "waitpid");
307  }
308  ok = 0;
309  if (WIFSIGNALED(status)) {
310  printf("Signal %d\n", WTERMSIG(status));
311  if (WTERMSIG(status) == ops[opindex].sig) {
312  ok = 1;
313  }
314  }
315  else {
316  printf("Exit %d\n", WEXITSTATUS(status));
317  if (WEXITSTATUS(status) == MAGIC) {
318  ok = 1;
319  }
320  }
321  if (ok) {
322  printf("Ok.\n");
323  }
324  else {
325  printf("FAILED: expected signal %d\n",
326  ops[opindex].sig);
327  }
328  printf("\n");
329  return;
330  }
331  }
332  /* child, or not forking */
333 
334  ops[opindex].f();
335 
336  if (op == 'f') {
337  warnx(".... I guess you don't support read-only segments");
338  /* use this magic signaling value so parent doesn't say FAIL */
339  _exit(MAGIC);
340  }
341  errx(1, "I wasn't killed!");
342 }

345 void
346 ask(void)
347 {
348  unsigned i;
349  int op;
350 
351  while (1) {
352 
353  for (i=0; ops[i].name; i++) {
354  printf("[%c] %s\n", ops[i].ch, ops[i].name);
355  }
356  printf("[-] Disable forking\n");
357  printf("[+] Enable forking (default)\n");
358  printf("[*] Run everything\n");
359  printf("[!] Quit\n");
360 
361  printf("Choose: ");
362  op = getchar();
363 
364  if (op == '!') {
365  break;
366  }
367 
368  runop(op);
369  }
370 }

372 int
373 main(int argc, char **argv)
374 {
375  if (argc == 0 || argc == 1) {
376  /* no arguments */
377  ask();
378  }
379  else {
380  /* run the selected ops */
381  for (int i=1; i<argc; i++) {
382  for (size_t j=0; argv[i][j]; j++) {
383  runop(argv[i][j]);
384  }
385  }
386  }
387  return 0;
