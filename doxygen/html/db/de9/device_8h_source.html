54 /*
55  * Device operations.
56  * devop_eachopen - called on each open call to allow denying the open
57  * devop_io - for both reads and writes (the uio indicates the direction)
58  * devop_ioctl - miscellaneous control operations
59  */
60 struct device_ops {
61  int (*devop_eachopen)(struct device *, int flags_from_open);
62  int (*devop_io)(struct device *, struct uio *);
63  int (*devop_ioctl)(struct device *, int op, userptr_t data);
64 };
