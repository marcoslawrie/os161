38 ntohll(uint64_t x)
39 {
40  uint32_t x0, x1, y0, y1;
41 
42  if (ntohl(1) == 1) {
43  return x;
44  }
45 
46  x0 = x & 0xffffffff;
47  y0 = ntohl(x0);
48  x1 = x >> 32;
49  y1 = ntohl(x1);
50  return ((uint64_t)y0 << 32) | y1;
51 }
