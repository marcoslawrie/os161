51 static
52 int
53 geti(void)
54 {
55  int val=0;
56  int ch, digits=0;
57 
58  while (1) {
59  ch = getchar();
60  if (ch=='\n' || ch=='\r') {
61  putchar('\n');
62  break;
63  }
64  else if ((ch=='\b' || ch==127) && digits>0) {
65  printf("\b \b");
66  val = val/10;
67  digits--;
68  }
69  else if (ch>='0' && ch<='9') {
70  putchar(ch);
71  val = val*10 + (ch-'0');
72  digits++;
73  }
74  else {
75  putchar('\a');
76  }
77  }
78 
79  if (digits==0) {
80  return -1;
81  }
82  return val;
83 }

87 getseed(void)
88 {
89  int fd, len;
90  unsigned long seed;
91 
92  fd = open(_PATH_RANDOM, O_RDONLY);
93  if (fd < 0) {
94  err(1, "%s", _PATH_RANDOM);
95  }
96  len = read(fd, &seed, sizeof(seed));
97  if (len < 0) {
98  err(1, "%s", _PATH_RANDOM);
99  }
100  else if (len < (int)sizeof(seed)) {
101  errx(1, "%s: Short read", _PATH_RANDOM);
102  }
103  close(fd);
104 
105  return seed;
106 }

121 static
122 void
123 dowait(pid_t pid)
124 {
125  int status;
126  int result;
127 
128  result = waitpid(pid, &status, 0);
129  if (result == -1) {
130  err(1, "waitpid");
131  }
132  if (WIFSIGNALED(status)) {
133  errx(1, "child: Signal %d", WTERMSIG(status));
134  }
135  if (WIFEXITED(status) && WEXITSTATUS(status) != 0) {
136  errx(1, "child: Exit %d", WEXITSTATUS(status));
137  }

151 /*
152  * Fill a page of memory with a test pattern.
153  */
154 static
155 void
156 markpage(volatile void *baseptr, unsigned pageoffset)
157 {
158  volatile char *pageptr;
159  size_t n, i;
160  volatile unsigned long *pl;
161  unsigned long val;
162 
163  pageptr = baseptr;
164  pageptr += (size_t)PAGE_SIZE * pageoffset;
165 
166  pl = (volatile unsigned long *)pageptr;
167  n = PAGE_SIZE / sizeof(unsigned long);
168 
169  for (i=0; i<n; i++) {
170  val = ((unsigned long)i ^ (unsigned long)pageoffset);
171  pl[i] = val;
172  }
173 }

175 /*
176  * Check a page marked with markblock()
177  */
178 static
179 int
180 checkpage(volatile void *baseptr, unsigned pageoffset, bool neednl)
181 {
182  volatile char *pageptr;
183  size_t n, i;
184  volatile unsigned long *pl;
185  unsigned long val;
186 
187  pageptr = baseptr;
188  pageptr += (size_t)PAGE_SIZE * pageoffset;
189 
190  pl = (volatile unsigned long *)pageptr;
191  n = PAGE_SIZE / sizeof(unsigned long);
192 
193  for (i=0; i<n; i++) {
194  val = ((unsigned long)i ^ (unsigned long)pageoffset);
195  if (pl[i] != val) {
196  if (neednl) {
197  printf("\n");
198  }
199  printf("FAILED: data mismatch at offset %lu of page "
200  "at 0x%lx: %lu vs. %lu\n",
201  (unsigned long) (i*sizeof(unsigned long)),
202  (unsigned long)(uintptr_t)pl,
203  pl[i], val);
204  return -1;
205  }
206  }
207 
208  return 0;
209 }

231 static
232 int
233 checkpagelight(volatile void *baseptr, unsigned pageoffset, bool neednl)
234 {
235  volatile char *pageptr;
236  volatile unsigned long *pl;
237 
238  pageptr = baseptr;
239  pageptr += (size_t)PAGE_SIZE * pageoffset;
240 
241  pl = (volatile unsigned long *)pageptr;
242  if (pl[0] != pageoffset) {
243  if (neednl) {
244  printf("\n");
245  }
246  printf("FAILED: data mismatch at offset 0 of page "
247  "at 0x%lx: %lu vs. %u\n",
248  (unsigned long)(uintptr_t)pl,
249  pl[0], pageoffset);
250  return -1;
251  }
252  return 0;

277 static
278 void
279 setup(void)
280 {
281  void *op;
282  uintptr_t opx;
283  size_t amount;
284  int error;
285 
286  op = dosbrk(0);
287  opx = (uintptr_t)op;
288 
289  if (opx % PAGE_SIZE) {
290  amount = PAGE_SIZE - (opx % PAGE_SIZE);
291  if (sbrk(amount) == (void *)-1) {
292  error = errno;
293  warnx("Initial heap was not page aligned");
294  warnx("...and trying to align it gave: %s",
295  strerror(error));
296  }
297  }
298 
299  op = dosbrk(0);
300  opx = (uintptr_t)op;
301 
302  if (opx % PAGE_SIZE) {
303  warnx("Initial heap was not page aligned");
304  errx(1, "...and trying to align it didn't take.");
305  }
306 }

555 // heap size

557 /*
558  * Allocate all memory at once.
559  *
560  * This works by trying successively smaller sizes until one succeeds.
561  * Note that if you allow arbitrary swap overcommit this test will run
562  * the system out of swap. If this kills the system that's probably ok
563  * (check with your course staff, but handing OOM is difficult and
564  * probably not worth the time it would take you)... but ideally no
565  * one process is allowed to get big enough to do this.
566  *
567  * The recommended behavior is to set the process's maximum heap size
568  * to the amount of available RAM + swap that can be used at once.
569  * Because the test uses powers of two, this doesn't have to be very
570  * precise (e.g. counting the size of the non-heap parts of the
571  * process probably doesn't matter unless you're very unlucky) and
572  * isn't very difficult.
573  *
574  * Another option is to disallow swap overcommit, in which case you
575  * have sbrk try to reserve swap pages for each allocation, which will
576  * fail for huge allocations. The problem with this in practice is
577  * that you end up needing a huge swap disk even to run relatively
578  * small workloads, and then this test takes forever to run.
579  */
580 static
581 void
582 test9(void)
583 {
584  size_t size;
585  unsigned i, pages, dot;
586  void *p;
587  bool bad;
588 
589 #define HUGESIZE (1024 * 1024 * 1024) /* 1G */
590 
591  printf("Checking how much memory we can allocate:\n");
592  for (size = HUGESIZE; (p = sbrk(size)) == (void *)-1; size = size/2) {
593  printf(" %9lu bytes: failed\n", (unsigned long) size);
594  }
595  printf(" %9lu bytes: succeeded\n", (unsigned long) size);
596  printf("Passed sbrk test 9 (part 1/5)\n");
597 
598  printf("Touching each page.\n");
599  pages = size / PAGE_SIZE;
600  dot = pages / 64;
601  for (i=0; i<pages; i++) {
602  markpagelight(p, i);
603  if (dot > 0 && i % dot == 0) {
604  printf(".");
605  }
606  }
607  if (dot > 0) {
608  printf("\n");
609  }
610 
611  printf("Testing each page.\n");
612  bad = false;
613  for (i=0; i<pages; i++) {
614  if (checkpagelight(p, i, dot > 0)) {
615  if (dot > 0) {
616  printf("\n");
617  }
618  warnx("FAILED: data corrupt");
619  bad = true;
620  }
621  if (dot > 0 && i % dot == 0) {
622  printf(".");
623  }
624  }
625  if (dot > 0) {
626  printf("\n");
627  }
628  if (bad) {
629  exit(1);
630  }
631  printf("Passed sbrk test 9 (part 2/5)\n");
632 
633  printf("Freeing the memory.\n");
634  (void)dosbrk(-size);
635  printf("Passed sbrk test 9 (part 3/5)\n");
636 
637  printf("Allocating the memory again.\n");
638  (void)dosbrk(size);
639  printf("Passed sbrk test 9 (part 4/5)\n");
640 
641  printf("And really freeing it.\n");
642  (void)dosbrk(-size);
643  printf("Passed sbrk test 9 (all)\n");
644 }

957 void
958 stresstest(unsigned long seed, bool large)
959 {
960  const unsigned loops = 10000;
961  const unsigned dot = 200;
962 
963  void *op;
964  unsigned num;
965  unsigned i, j, pages;
966  unsigned long r;
967  bool bad, neg;
968 
969  srandom(seed);
970  printf("Seeded random number generator with %lu.\n", seed);
971 
972  op = dosbrk(0);
973 
974  bad = false;
975  num = 0;
976  for (i=0; i<loops; i++) {
977  /*
978  * The goal of this test is not to stress the VM system
979  * by thrashing swap (other tests do that) but by running
980  * the sbrk code a lot. So, clamp the total size at either
981  * 128K or 512K, which is 32 or 128 pages respectively.
982  */
983  r = random();
984  pages = r % (large ? 32 : 8);
985  neg = pages <= num && ((r & 128) == 0);
986  if (!neg && num + pages > (large ? 128 : 32)) {
987  neg = 1;
988  }
989  if (neg) {
990  dosbrk(-(pages * PAGE_SIZE));
991  num -= pages;
992  }
993  else {
994  dosbrk(pages * PAGE_SIZE);
995  for (j=0; j<pages; j++) {
996  markpagelight(op, num + j);
997  }
998  num += pages;
999  }
1000  for (j=0; j<num; j++) {
1001  if (checkpagelight(op, j, true)) {
1002  printf("\n");
1003  warnx("FAILED: data corrupt on page %u", j);
1004  bad = true;
1005  }
1006  }
1007  if (i % dot == 0) {
1008  printf(".");
1009  }
1010  }
1011  printf("\n");
1012  if (bad) {
1013  warnx("FAILED");
1014  exit(1);
1015  }
1016 
1017  dosbrk(-(num * PAGE_SIZE));
1018  printf("Passed sbrk %s stress test.\n", large ? "large" : "small");
1019 }
