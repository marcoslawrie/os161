33 /*
34  * Indirect block access macros
35  *
36  * These are macros for working with the direct and indirect block
37  * pointers in the inode. The scheme here supports a range of possible
38  * configurations, because sometimes adding large file support to SFS
39  * is part of an assignment; there is and should be no obligation to
40  * pick any particular layout, and we'd like sfsck to build and run
41  * seamlessly provided that the values declared in kern/sfs.h are
42  * correct.
43  *
44  * SFS_NDIRECT, SFS_NINDIRECT, SFS_NDINDIRECT, and SFS_NTINDIRECT
45  * should always be defined. If zero, no corresponding field is
46  * assumed to exist in the inode. If one, the field is assumed to
47  * be a single value and not an array. If greater than one, the
48  * field is assumed to be an array.
49  */
50 
51 #ifndef SFS_NDIRECT
52 #error "SFS_NDIRECT not defined"
53 #endif
54 #ifndef SFS_NINDIRECT
55 #error "SFS_NINDIRECT not defined"
56 #endif
57 #ifndef SFS_NDINDIRECT
58 #error "SFS_NDINDIRECT not defined"
59 #endif
60 #ifndef SFS_NTINDIRECT
61 #error "SFS_NTINDIRECT not defined"
62 #endif

64 /*
65  * For x in D, I, II, III (direct, indirect, 2x/3x indirect) we
66  * provide:
67  *
68  * NUM_x the number of blocks of this type
69  * GET_x retrieve the i'th block of this type
70  * SET_x lvalue for the i'th block of this type
71  * RANGE_x size of the block range mapped with one block of this type
72  * INOMAX_x maximum block number mapped by using this type in the inode
73  *
74  * It is important that the accessor macros (SET_x/GET_x) not refer to
75  * a nonexistent field of the inode in the case where there are zero
76  * blocks of that type, as that will lead to compile failure. Hence the
77  * lengthy definitions. So far I haven't thought any useful cpp hackery
78  * to make them more concise.
79  */
80 
81 /* numbers */
82 
83 #define NUM_D SFS_NDIRECT
84 #define NUM_I SFS_NINDIRECT
85 #define NUM_II SFS_NDINDIRECT
86 #define NUM_III SFS_NTINDIRECT
87 
88 /* blocks */
89 
90 #if NUM_D == 0
91 #define GET_D(sfi, i) GET0_x(sfi, sfi_direct, i)
92 #define SET_D(sfi, i) SET0_x(sfi, sfi_direct, i)
93 #elif NUM_D == 1
94 #define GET_D(sfi, i) GET1_x(sfi, sfi_direct, i)
95 #define SET_D(sfi, i) SET1_x(sfi, sfi_direct, i)
96 #else
97 #define GET_D(sfi, i) GETN_x(sfi, sfi_direct, i)
98 #define SET_D(sfi, i) SETN_x(sfi, sfi_direct, i)
99 #endif
100 
101 #if NUM_I == 0

104 #elif NUM_I == 1
105 #define GET_I(sfi, i) GET1_x(sfi, sfi_indirect, i)
106 #define SET_I(sfi, i) SET1_x(sfi, sfi_indirect, i)
107 #else

109 #define SET_I(sfi, i) SETN_x(sfi, sfi_indirect, i)
110 #endif

134 /* the generic forms of the block macros */
135 
136 #define GET0_x(sfi, field, i) ((void)(i), (void)(sfi), 0)
137 #define GET1_x(sfi, field, i) ((void)(i), (sfi)->field)
138 #define GETN_x(sfi, field, i) ((sfi)->field[(i)])
139 
140 #define SET0_x(sfi, field, i) (*((void)(i), (void)(sfi), (uint32_t *)NULL))
141 #define SET1_x(sfi, field, i) (*((void)(i), &(sfi)->field))
142 #define SETN_x(sfi, field, i) ((sfi)->field[(i)])
