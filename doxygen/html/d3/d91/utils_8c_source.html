53 /*
54  * Wrapper around realloc. OSZ is the old block size, which we need if
55  * we're going to emulate realloc with malloc.
56  */
57 void *
58 dorealloc(void *op, size_t osz, size_t nsz)
59 {
60  void *np;
61 #ifdef NO_REALLOC
62  size_t copysz;
63 
64  np = domalloc(nsz);
65  if (op != NULL) {
66  copysz = osz < nsz ? osz : nsz;
67  memcpy(np, op, copysz);
68  free(op);
69  }
70 #else
71  (void)osz;
72  np = realloc(op, nsz);
73  if (np == NULL) {
74  errx(EXIT_FATAL, "Out of memory");
75  }
76 #endif
77  return np;
78 }

91 /*
92  * Check if BUF, a string field of length MAXLEN, contains a null
93  * terminator. If not, slam one in and return 1.
94  */
95 int
96 checknullstring(char *buf, size_t maxlen)
97 {
98  size_t i;
99  for (i=0; i<maxlen; i++) {
100  if (buf[i]==0) {
101  return 0;
102  }
103  }
104  buf[maxlen-1] = 0;
105  return 1;
106 }

108 /*
109  * Check if BUF contains characters not allowed in file and volume
110  * names. If so, stomp them and return 1.
111  */
112 int
113 checkbadstring(char *buf)
114 {
115  size_t i;
116  int rv = 0;
117 
118  for (i=0; buf[i]; i++) {
119  if (buf[i]==':' || buf[i]=='/') {
120  buf[i] = '_';
121  rv = 1;
122  }
123  }
124  return rv;
125 }
