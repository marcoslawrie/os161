47 /*
48  * Load the superblock.
49  */
50 void
51 sb_load(void)
52 {
53  sfs_readsb(SFS_SUPER_BLOCK, &sb);
54  if (sb.sb_magic != SFS_MAGIC) {
55  errx(EXIT_FATAL, "Not an sfs filesystem");
56  }
57 
58  assert(sb.sb_nblocks > 0);
59  assert(SFS_FREEMAPBLOCKS(sb.sb_nblocks) > 0);
60 }

62 /*
63  * Validate the superblock.
64  */
65 void
66 sb_check(void)
67 {
68  int schanged=0;
69 
70  /*
71  * FUTURE: should we check sb.sb_nblocks against diskblocks()?
72  */
73 
74  /* Check the superblock fields */
75 
76  if (checknullstring(sb.sb_volname, sizeof(sb.sb_volname))) {
77  warnx("Volume name not null-terminated (fixed)");
78  setbadness(EXIT_RECOV);
79  schanged = 1;
80  }
81  if (checkbadstring(sb.sb_volname)) {
82  warnx("Volume name contains illegal characters (fixed)");
83  setbadness(EXIT_RECOV);
84  schanged = 1;
85  }
86  if (checkzeroed(sb.reserved, sizeof(sb.reserved))) {
87  warnx("Reserved section of superblock not zeroed (fixed)");
88  setbadness(EXIT_RECOV);
89  schanged = 1;
90  }
91 
92  /* Write the superblock back if necessary */
93  if (schanged) {
94  sfs_writesb(SFS_SUPER_BLOCK, &sb);
95  }
96 }
