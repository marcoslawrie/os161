68 int
69 lrandom_read(void *devdata, struct uio *uio)
70 {
71  struct lrandom_softc *lr = devdata;
72  uint32_t val;
73  int result;
74 
75  while (uio->uio_resid > 0) {
76  val = bus_read_register(lr->lr_bus, lr->lr_buspos,
77  LR_REG_RAND);
78  result = uiomove(&val, sizeof(val), uio);
79  if (result) {
80  return result;
81  }
82  }
83 
84  return 0;
85 }
