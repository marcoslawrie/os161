171 static
172 unsigned
173 sizeblocks(enum sizes sz)
174 {
175  /*
176  * XXX for now hardwire things we know about SFS
177  */
178 #define BLOCKSIZE /*SFS_BLOCKSIZE*/ 512
179  static const unsigned ndb = /*SFS_NDIRECT*/ 15;
180  static const unsigned dbperidb = /*SFS_DBPERIDB*/ 128;
181 
182  switch (sz) {
183  case SIZE_ONE:
184  /* one block; 512 bytes */
185  return 1;
186  case SIZE_SMALL:
187  /* fits in direct blocks only; 7.5K */
188  return ndb;
189  case SIZE_MEDIUM:
190  /* uses an indirect block; ~40K */
191  return ndb + dbperidb / 2;
192  case SIZE_LARGE:
193  /* uses a double-indirect block; 4.2M */
194  return ndb + dbperidb + dbperidb * dbperidb / 2;
195  case SIZE_LARGEPLUS:
196  /* requires a triple-indirect block; 8.5M */
197  return ndb + dbperidb + dbperidb * dbperidb + dbperidb / 2;
198  }
199  assert(0);
200  return 12;
201 }

213 static
214 void
215 file_randomwrite(struct file *f, enum sizes sz,
216  unsigned startskip, unsigned endskip)
217 {
218  unsigned nblocks, nwrites, i;
219  unsigned blocknum;
220  off_t pos;
221 
222  nblocks = sizeblocks(sz);
223  assert(nblocks > startskip + endskip);
224 
225  nwrites = nblocks/6;
226  if (nwrites < 2) {
227  nwrites = 2;
228  }
229 
230  nblocks -= startskip + endskip;
231  for (i=0; i<nwrites; i++) {
232  blocknum = startskip + random() % nblocks;
233  pos = (off_t)BLOCKSIZE * blocknum;
234  op_write(f, pos, BLOCKSIZE);
235  }
236 }
