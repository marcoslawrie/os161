71  int ch, digits=0;
72 
73  while (1) {
74  ch = getchar();
75  if (ch=='\n' || ch=='\r') {
76  putchar('\n');
77  break;
78  }
79  else if ((ch=='\b' || ch==127) && digits>0) {
80  printf("\b \b");
81  val = val/10;
82  digits--;
83  }
84  else if (ch>='0' && ch<='9') {
85  putchar(ch);

87  digits++;
88  }
89  else {
90  putchar('\a');
91  }
92  }
93 
94  if (digits==0) {
95  return -1;
96  }
97  return val;
98 }

107 markblock(volatile void *ptr, size_t size, unsigned bias, int doprint)
108 {
109  size_t n, i;
110  unsigned long *pl;
111  unsigned long val;
112 
113  pl = (unsigned long *)ptr;
114  n = size / sizeof(unsigned long);
115 
116  for (i=0; i<n; i++) {
117  val = ((unsigned long)i ^ (unsigned long)bias);
118  pl[i] = val;
119  if (doprint && (i%64==63)) {
120  printf(".");
121  }
122  }
123  if (doprint) {
124  printf("\n");
125  }
126 }

128 /*
129  * Check a block marked with markblock()
130  */
131 static
132 int
133 checkblock(volatile void *ptr, size_t size, unsigned bias, int doprint)
134 {
135  size_t n, i;
136  unsigned long *pl;
137  unsigned long val;
138 
139  pl = (unsigned long *)ptr;
140  n = size / sizeof(unsigned long);
141 
142  for (i=0; i<n; i++) {
143  val = ((unsigned long)i ^ (unsigned long)bias);
144  if (pl[i] != val) {
145  if (doprint) {
146  printf("\n");
147  }
148  printf("FAILED: data mismatch at offset %lu of block "
149  "at 0x%lx: %lu vs. %lu\n",
150  (unsigned long) (i*sizeof(unsigned long)),
151  (unsigned long)(uintptr_t)pl,
152  pl[i], val);
153  return -1;
154  }
155  if (doprint && (i%64==63)) {
156  printf(".");
157  }
158  }
159  if (doprint) {
160  printf("\n");
161  }
162 
163  return 0;
164 }

203 /*
204  * Test 2
205  *
206  * Tests if malloc gracefully handles failing requests.
207  *
208  * This test assumes that one of the following conditions holds:
209  * 1. swap is not overcommitted; or
210  * 2. user processes are limited to some maximum size, and enough
211  * swap exists to hold a maximal user process.
212  *
213  * That is, it assumes that malloc returns NULL when out of memory,
214  * and that the process will not be killed for running out of
215  * memory/swap at other times.
216  *
217  * If mallocing more memory than the system can actually provide
218  * backing for succeeds, this test will blow up. That's ok, but please
219  * provide a way to switch on one of the above conditions so this test
220  * can be run.
221  *
222  * This test works by trying a huge malloc, and then trying
223  * successively smaller mallocs until one works. Then it touches the
224  * whole block to make sure the memory is actually successfully
225  * allocated. Then it frees the block and allocates it again, which
226  * should succeed.
227  *
228  * Note that this test may give spurious failures if anything else is
229  * running at the same time and changing the amount of memory
230  * available.
231  */
232

384 /*
385  * Test 4
386  *
387  * Tries to test in detail if malloc coalesces the free list properly.
388  *
389  * This test will likely fail if something other than a basic first-fit/
390  * next-fit/best-fit algorithm is used.
391  */
392 
393 static
394 void
395 test4(void)
396 {
397  void *x, *y, *z;
398  unsigned long lx, ly, lz, overhead, zsize;
399 
400  printf("Entering malloc test 4.\n");
401  printf("This test is intended for first/best-fit based mallocs.\n");
402  printf("This test may not work correctly if run after other tests.\n");
403 
404  printf("Testing free list coalescing:\n");
405 
406  x = malloc(SMALLSIZE);
407  if (x==NULL) {
408  printf("FAILED: malloc(%u) failed\n", SMALLSIZE);
409  return;
410  }
411 
412  y = malloc(MEDIUMSIZE);
413  if (y==NULL) {
414  printf("FAILED: malloc(%u) failed\n", MEDIUMSIZE);
415  return;
416  }
417 
418  if (sizeof(unsigned long) < sizeof(void *)) {
419  printf("Buh? I can't fit a void * in an unsigned long\n");
420  printf("ENVIRONMENT FAILED...\n");
421  return;
422  }
423 
424  lx = (unsigned long)x;
425  ly = (unsigned long)y;
426 
427  printf("x is 0x%lx; y is 0x%lx\n", lx, ly);
428 
429  /*
430  * The memory should look something like this:
431  *
432  * OXXXOYYYYYYYYYYY
433  *
434  * where O are optional overhead (indexing) blocks.
435  */
436 
437  /* This is obviously wrong. */
438  if (lx == ly) {
439  printf("FAIL: x == y\n");
440  return;
441  }
442 
443  /*
444  * Check for overlap. It is sufficient to check if the start
445  * of each block is within the other block. (If the end of a
446  * block is within the other block, either the start is too,
447  * or the other block's start is within the first block.)
448  */
449  if (lx < ly && lx + SMALLSIZE > ly) {
450  printf("FAIL: y starts within x\n");
451  return;
452  }
453  if (ly < lx && ly + MEDIUMSIZE > lx) {
454  printf("FAIL: x starts within y\n");
455  return;
456  }
457 
458  /*
459  * If y is lower than x, some memory scheme we don't
460  * understand is in use, or else there's already stuff on the
461  * free list.
462  */
463  if (ly < lx) {
464  printf("TEST UNSUITABLE: y is below x\n");
465  return;
466  }
467 
468  /*
469  * Compute the space used by index structures.
470  */
471  overhead = ly - (lx + SMALLSIZE);
472  printf("Apparent block overhead: %lu\n", overhead);
473 
474  if (overhead > ABSURD_OVERHEAD) {
475  printf("TEST UNSUITABLE: block overhead absurdly large.\n");
476  return;
477  }
478  if (overhead > OVERHEAD) {
479  printf("FAIL: block overhead is too large.\n");
480  return;
481  }
482 
483  printf("Freeing blocks...\n");
484  free(x);
485  free(y);
486 
487  zsize = SMALLSIZE + MEDIUMSIZE + overhead;
488 
489  printf("Now allocating %lu bytes... should reuse the space.\n", zsize);
490  z = malloc(zsize);
491  if (z == NULL) {
492  printf("FAIL: Allocation failed...\n");
493  return;
494  }
495 
496  lz = (unsigned long) z;
497 
498  printf("z is 0x%lx (x was 0x%lx, y 0x%lx)\n", lz, lx, ly);
499 
500  if (lz==lx) {
501  printf("Passed.\n");
502  }
503  else {
504  printf("Failed.\n");
505  }
506 
507  free(z);
508 }

560  ptrs[n] = NULL;
561  psizes[n] = 0;
562  }

564  printf(".");

568

573  }
