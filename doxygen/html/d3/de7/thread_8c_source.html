74 /*
75  * Stick a magic number on the bottom end of the stack. This will
76  * (sometimes) catch kernel stack overflows. Use thread_checkstack()
77  * to test this.
78  */
79 static
80 void
81 thread_checkstack_init(struct thread *thread)
82 {
83  ((uint32_t *)thread->t_stack)[0] = THREAD_STACK_MAGIC;
84  ((uint32_t *)thread->t_stack)[1] = THREAD_STACK_MAGIC;
85  ((uint32_t *)thread->t_stack)[2] = THREAD_STACK_MAGIC;
86  ((uint32_t *)thread->t_stack)[3] = THREAD_STACK_MAGIC;
87 }
88 
89 /*
90  * Check the magic number we put on the bottom end of the stack in
91  * thread_checkstack_init. If these assertions go off, it most likely
92  * means you overflowed your stack at some point, which can cause all
93  * kinds of mysterious other things to happen.
94  *
95  * Note that when ->t_stack is NULL, which is the case if the stack
96  * cannot be freed (which in turn is the case if the stack is the boot
97  * stack, and the thread is the boot thread) this doesn't do anything.
98  */
99 static
100 void
101 thread_checkstack(struct thread *thread)
102 {
103  if (thread->t_stack != NULL) {
104  KASSERT(((uint32_t*)thread->t_stack)[0] == THREAD_STACK_MAGIC);
105  KASSERT(((uint32_t*)thread->t_stack)[1] == THREAD_STACK_MAGIC);
106  KASSERT(((uint32_t*)thread->t_stack)[2] == THREAD_STACK_MAGIC);
107  KASSERT(((uint32_t*)thread->t_stack)[3] == THREAD_STACK_MAGIC);
108  }
109 }

111 /*
112  * Create a thread. This is used both to create a first thread
113  * for each CPU and to create subsequent forked threads.
114  */
115 static
116 struct thread *
117 thread_create(const char *name)
118 {
119  struct thread *thread;
120 
121  DEBUGASSERT(name != NULL);
122 
123  thread = kmalloc(sizeof(*thread));
124  if (thread == NULL) {
125  return NULL;
126  }
127 
128  thread->t_name = kstrdup(name);
129  if (thread->t_name == NULL) {
130  kfree(thread);
131  return NULL;
132  }
133  thread->t_wchan_name = "NEW";
134  thread->t_state = S_READY;
135 
136  /* Thread subsystem fields */
137  thread_machdep_init(&thread->t_machdep);
138  threadlistnode_init(&thread->t_listnode, thread);
139  thread->t_stack = NULL;
140  thread->t_context = NULL;
141  thread->t_cpu = NULL;
142  thread->t_proc = NULL;
143  HANGMAN_ACTORINIT(&thread->t_hangman, thread->t_name);
144 
145  /* Interrupt state fields */
146  thread->t_in_interrupt = false;
147  thread->t_curspl = IPL_HIGH;
148  thread->t_iplhigh_count = 1; /* corresponding to t_curspl */
149 
150  /* If you add to struct thread, be sure to initialize here */
151 
152  return thread;
153 }

155 /*
156  * Create a CPU structure. This is used for the bootup CPU and
157  * also for secondary CPUs.
158  *
159  * The hardware number (the number assigned by firmware or system
160  * board config or whatnot) is tracked separately because it is not
161  * necessarily anything sane or meaningful.
162  */
163 struct cpu *
164 cpu_create(unsigned hardware_number)
165 {
166  struct cpu *c;
167  int result;
168  char namebuf[16];
169 
170  c = kmalloc(sizeof(*c));
171  if (c == NULL) {
172  panic("cpu_create: Out of memory\n");
173  }
174 
175  c->c_self = c;
176  c->c_hardware_number = hardware_number;
177 
178  c->c_curthread = NULL;
179  threadlist_init(&c->c_zombies);
180  c->c_hardclocks = 0;
181  c->c_spinlocks = 0;
182 
183  c->c_isidle = false;
184  threadlist_init(&c->c_runqueue);
185  spinlock_init(&c->c_runqueue_lock);
186 
187  c->c_ipi_pending = 0;
188  c->c_numshootdown = 0;
189  spinlock_init(&c->c_ipi_lock);
190 
191  result = cpuarray_add(&allcpus, c, &c->c_number);
192  if (result != 0) {
193  panic("cpu_create: array_add: %s\n", strerror(result));
194  }
195 
196  snprintf(namebuf, sizeof(namebuf), "<boot #%d>", c->c_number);
197  c->c_curthread = thread_create(namebuf);
198  if (c->c_curthread == NULL) {
199  panic("cpu_create: thread_create failed\n");
200  }
201  c->c_curthread->t_cpu = c;
202 
203  if (c->c_number == 0) {
204  /*
205  * Leave c->c_curthread->t_stack NULL for the boot
206  * cpu. This means we're using the boot stack, which
207  * can't be freed. (Exercise: what would it take to
208  * make it possible to free the boot stack?)
209  */
210  /*c->c_curthread->t_stack = ... */
211  }
212  else {
213  c->c_curthread->t_stack = kmalloc(STACK_SIZE);
214  if (c->c_curthread->t_stack == NULL) {
215  panic("cpu_create: couldn't allocate stack");
216  }
217  thread_checkstack_init(c->c_curthread);
218  }
219 
220  /*
221  * If there is no curcpu (or curthread) yet, we are creating
222  * the first (boot) cpu. Initialize curcpu and curthread as
223  * early as possible so that other code can take locks without
224  * exploding.
225  */
226  if (!CURCPU_EXISTS()) {
227  /*
228  * Initializing curcpu and curthread is
229  * machine-dependent because either of curcpu and
230  * curthread might be defined in terms of the other.
231  */
232  INIT_CURCPU(c, c->c_curthread);
233 
234  /*
235  * Now make sure both t_cpu and c_curthread are
236  * set. This might be partially redundant with
237  * INIT_CURCPU depending on how things are defined.
238  */
239  curthread->t_cpu = curcpu;
240  curcpu->c_curthread = curthread;
241  }
242 
243  HANGMAN_ACTORINIT(&c->c_hangman, "cpu");
244 
245  result = proc_addthread(kproc, c->c_curthread);
246  if (result) {
247  panic("cpu_create: proc_addthread:: %s\n", strerror(result));
248  }
249 
250  cpu_machdep_init(c);
251 
252  return c;
253 }

255 /*
256  * Destroy a thread.
257  *
258  * This function cannot be called in the victim thread's own context.
259  * Nor can it be called on a running thread.
260  *
261  * (Freeing the stack you're actually using to run is ... inadvisable.)
262  */
263 static
264 void
265 thread_destroy(struct thread *thread)
266 {
267  KASSERT(thread != curthread);
268  KASSERT(thread->t_state != S_RUN);
269 
270  /*
271  * If you add things to struct thread, be sure to clean them up
272  * either here or in thread_exit(). (And not both...)
273  */
274 
275  /* Thread subsystem fields */
276  KASSERT(thread->t_proc == NULL);
277  if (thread->t_stack != NULL) {
278  kfree(thread->t_stack);
279  }
280  threadlistnode_cleanup(&thread->t_listnode);
281  thread_machdep_cleanup(&thread->t_machdep);
282 
283  /* sheer paranoia */
284  thread->t_wchan_name = "DESTROYED";
285 
286  kfree(thread->t_name);
287  kfree(thread);
288 }

290 /*
291  * Clean up zombies. (Zombies are threads that have exited but still
292  * need to have thread_destroy called on them.)
293  *
294  * The list of zombies is per-cpu.
295  */
296 static
297 void
298 exorcise(void)
299 {
300  struct thread *z;
301 
302  while ((z = threadlist_remhead(&curcpu->c_zombies)) != NULL) {
303  KASSERT(z != curthread);
304  KASSERT(z->t_state == S_ZOMBIE);
305  thread_destroy(z);
306  }
307 }

309 /*
310  * On panic, stop the thread system (as much as is reasonably
311  * possible) to make sure we don't end up letting any other threads
312  * run.
313  */
314 void
315 thread_panic(void)
316 {
317  /*
318  * Kill off other CPUs.
319  *
320  * We could wait for them to stop, except that they might not.
321  */
322  ipi_broadcast(IPI_PANIC);
323 
324  /*
325  * Drop runnable threads on the floor.
326  *
327  * Don't try to get the run queue lock; we might not be able
328  * to. Instead, blat the list structure by hand, and take the
329  * risk that it might not be quite atomic.
330  */
331  curcpu->c_runqueue.tl_count = 0;
332  curcpu->c_runqueue.tl_head.tln_next = &curcpu->c_runqueue.tl_tail;
333  curcpu->c_runqueue.tl_tail.tln_prev = &curcpu->c_runqueue.tl_head;
334 
335  /*
336  * Ideally, we want to make sure sleeping threads don't wake
337  * up and start running. However, there's no good way to track
338  * down all the wchans floating around the system. Another
339  * alternative would be to set a global flag to make the wchan
340  * wakeup operations do nothing; but that would mean we
341  * ourselves couldn't sleep to wait for an I/O completion
342  * interrupt, and we'd like to be able to do that if the
343  * system isn't that badly hosed.
344  *
345  * So, do nothing else here.
346  *
347  * This may prove inadequate in practice and further steps
348  * might be needed. It may also be necessary to go through and
349  * forcibly unlock all locks or the like...
350  */
351 }

368 /*
369  * Thread system initialization.
370  */
371 void
372 thread_bootstrap(void)
373 {
374  cpuarray_init(&allcpus);
375 
376  /*
377  * Create the cpu structure for the bootup CPU, the one we're
378  * currently running on. Assume the hardware number is 0; that
379  * might be updated later by mainbus-type code. This also
380  * creates a thread structure for the first thread, the one
381  * that's already implicitly running when the kernel is
382  * started from the bootloader.
383  */
384  KASSERT(CURCPU_EXISTS() == false);
385  (void)cpu_create(0);
386  KASSERT(CURCPU_EXISTS() == true);
387 
388  /* cpu_create() should also have set t_proc. */
389  KASSERT(curcpu != NULL);
390  KASSERT(curthread != NULL);
391  KASSERT(curthread->t_proc != NULL);
392  KASSERT(curthread->t_proc == kproc);
393 
394  /* Done */
395 }

397 /*
398  * New CPUs come here once MD initialization is finished. curthread
399  * and curcpu should already be initialized.
400  *
401  * Other than clearing thread_start_cpus() to continue, we don't need
402  * to do anything. The startup thread can just exit; we only need it
403  * to be able to get into thread_switch() properly.
404  */
405 void
406 cpu_hatch(unsigned software_number)
407 {
408  char buf[64];
409 
410  KASSERT(curcpu != NULL);
411  KASSERT(curthread != NULL);
412  KASSERT(curcpu->c_number == software_number);
413 
414  spl0();
415  cpu_identify(buf, sizeof(buf));
416 
417  kprintf("cpu%u: %s\n", software_number, buf);
418 
419  V(cpu_startup_sem);
420  thread_exit();
421 }

423 /*
424  * Start up secondary cpus. Called from boot().
425  */
426 void
427 thread_start_cpus(void)
428 {
429  char buf[64];
430  unsigned i;
431 
432  cpu_identify(buf, sizeof(buf));
433  kprintf("cpu0: %s\n", buf);
434 
435  cpu_startup_sem = sem_create("cpu_hatch", 0);
436  mainbus_start_cpus();
437 
438  for (i=0; i<cpuarray_num(&allcpus) - 1; i++) {
439  P(cpu_startup_sem);
440  }
441  sem_destroy(cpu_startup_sem);
442  cpu_startup_sem = NULL;
443 }

445 /*
446  * Make a thread runnable.
447  *
448  * targetcpu might be curcpu; it might not be, too.
449  */
450 static
451 void
452 thread_make_runnable(struct thread *target, bool already_have_lock)
453 {
454  struct cpu *targetcpu;
455 
456  /* Lock the run queue of the target thread's cpu. */
457  targetcpu = target->t_cpu;
458 
459  if (already_have_lock) {
460  /* The target thread's cpu should be already locked. */
461  KASSERT(spinlock_do_i_hold(&targetcpu->c_runqueue_lock));
462  }
463  else {
464  spinlock_acquire(&targetcpu->c_runqueue_lock);
465  }
466 
467  /* Target thread is now ready to run; put it on the run queue. */
468  target->t_state = S_READY;
469  threadlist_addtail(&targetcpu->c_runqueue, target);
470 
471  if (targetcpu->c_isidle && targetcpu != curcpu->c_self) {
472  /*
473  * Other processor is idle; send interrupt to make
474  * sure it unidles.
475  */
476  ipi_send(targetcpu, IPI_UNIDLE);
477  }
478 
479  if (!already_have_lock) {
480  spinlock_release(&targetcpu->c_runqueue_lock);
481  }
482 }

484 /*
485  * Create a new thread based on an existing one.
486  *
487  * The new thread has name NAME, and starts executing in function
488  * ENTRYPOINT. DATA1 and DATA2 are passed to ENTRYPOINT.
489  *
490  * The new thread is created in the process P. If P is null, the
491  * process is inherited from the caller. It will start on the same CPU
492  * as the caller, unless the scheduler intervenes first.
493  */
494 int
495 thread_fork(const char *name,
496  struct proc *proc,
497  void (*entrypoint)(void *data1, unsigned long data2),
498  void *data1, unsigned long data2)
499 {
500  struct thread *newthread;
501  int result;
502 
503  newthread = thread_create(name);
504  if (newthread == NULL) {
505  return ENOMEM;
506  }
507 
508  /* Allocate a stack */
509  newthread->t_stack = kmalloc(STACK_SIZE);
510  if (newthread->t_stack == NULL) {
511  thread_destroy(newthread);
512  return ENOMEM;
513  }
514  thread_checkstack_init(newthread);
515 
516  /*
517  * Now we clone various fields from the parent thread.
518  */
519 
520  /* Thread subsystem fields */
521  newthread->t_cpu = curthread->t_cpu;
522 
523  /* Attach the new thread to its process */
524  if (proc == NULL) {
525  proc = curthread->t_proc;
526  }
527  result = proc_addthread(proc, newthread);
528  if (result) {
529  /* thread_destroy will clean up the stack */
530  thread_destroy(newthread);
531  return result;
532  }
533 
534  /*
535  * Because new threads come out holding the cpu runqueue lock
536  * (see notes at bottom of thread_switch), we need to account
537  * for the spllower() that will be done releasing it.
538  */
539  newthread->t_iplhigh_count++;
540 
541  /* Set up the switchframe so entrypoint() gets called */
542  switchframe_init(newthread, entrypoint, data1, data2);
543 
544  /* Lock the current cpu's run queue and make the new thread runnable */
545  thread_make_runnable(newthread, false);
546 
547  return 0;
548 }

550 /*
551  * High level, machine-independent context switch code.
552  *
553  * The current thread is queued appropriately and its state is changed
554  * to NEWSTATE; another thread to run is selected and switched to.
555  *
556  * If NEWSTATE is S_SLEEP, the thread is queued on the wait channel
557  * WC, protected by the spinlock LK. Otherwise WC and Lk should be
558  * NULL.
559  */
560 static
561 void
562 thread_switch(threadstate_t newstate, struct wchan *wc, struct spinlock *lk)
563 {
564  struct thread *cur, *next;
565  int spl;
566 
567  DEBUGASSERT(curcpu->c_curthread == curthread);
568  DEBUGASSERT(curthread->t_cpu == curcpu->c_self);
569 
570  /* Explicitly disable interrupts on this processor */
571  spl = splhigh();
572 
573  cur = curthread;
574 
575  /*
576  * If we're idle, return without doing anything. This happens
577  * when the timer interrupt interrupts the idle loop.
578  */
579  if (curcpu->c_isidle) {
580  splx(spl);
581  return;
582  }
583 
584  /* Check the stack guard band. */
585  thread_checkstack(cur);
586 
587  /* Lock the run queue. */
588  spinlock_acquire(&curcpu->c_runqueue_lock);
589 
590  /* Micro-optimization: if nothing to do, just return */
591  if (newstate == S_READY && threadlist_isempty(&curcpu->c_runqueue)) {
592  spinlock_release(&curcpu->c_runqueue_lock);
593  splx(spl);
594  return;
595  }
596 
597  /* Put the thread in the right place. */
598  switch (newstate) {
599  case S_RUN:
600  panic("Illegal S_RUN in thread_switch\n");
601  case S_READY:
602  thread_make_runnable(cur, true /*have lock*/);
603  break;
604  case S_SLEEP:
605  cur->t_wchan_name = wc->wc_name;
606  /*
607  * Add the thread to the list in the wait channel, and
608  * unlock same. To avoid a race with someone else
609  * calling wchan_wake*, we must keep the wchan's
610  * associated spinlock locked from the point the
611  * caller of wchan_sleep locked it until the thread is
612  * on the list.
613  */
614  threadlist_addtail(&wc->wc_threads, cur);
615  spinlock_release(lk);
616  break;
617  case S_ZOMBIE:
618  cur->t_wchan_name = "ZOMBIE";
619  threadlist_addtail(&curcpu->c_zombies, cur);
620  break;
621  }
622  cur->t_state = newstate;
623 
624  /*
625  * Get the next thread. While there isn't one, call cpu_idle().
626  * curcpu->c_isidle must be true when cpu_idle is
627  * called. Unlock the runqueue while idling too, to make sure
628  * things can be added to it.
629  *
630  * Note that we don't need to unlock the runqueue atomically
631  * with idling; becoming unidle requires receiving an
632  * interrupt (either a hardware interrupt or an interprocessor
633  * interrupt from another cpu posting a wakeup) and idling
634  * *is* atomic with respect to re-enabling interrupts.
635  *
636  * Note that c_isidle becomes true briefly even if we don't go
637  * idle. However, because one is supposed to hold the runqueue
638  * lock to look at it, this should not be visible or matter.
639  */
640 
641  /* The current cpu is now idle. */
642  curcpu->c_isidle = true;
643  do {
644  next = threadlist_remhead(&curcpu->c_runqueue);
645  if (next == NULL) {
646  spinlock_release(&curcpu->c_runqueue_lock);
647  cpu_idle();
648  spinlock_acquire(&curcpu->c_runqueue_lock);
649  }
650  } while (next == NULL);
651  curcpu->c_isidle = false;
652 
653  /*
654  * Note that curcpu->c_curthread may be the same variable as
655  * curthread and it may not be, depending on how curthread and
656  * curcpu are defined by the MD code. We'll assign both and
657  * assume the compiler will optimize one away if they're the
658  * same.
659  */
660  curcpu->c_curthread = next;
661  curthread = next;
662 
663  /* do the switch (in assembler in switch.S) */
664  switchframe_switch(&cur->t_context, &next->t_context);
665 
666  /*
667  * When we get to this point we are either running in the next
668  * thread, or have come back to the same thread again,
669  * depending on how you look at it. That is,
670  * switchframe_switch returns immediately in another thread
671  * context, which in general will be executing here with a
672  * different stack and different values in the local
673  * variables. (Although new threads go to thread_startup
674  * instead.) But, later on when the processor, or some
675  * processor, comes back to the previous thread, it's also
676  * executing here with the *same* value in the local
677  * variables.
678  *
679  * The upshot, however, is as follows:
680  *
681  * - The thread now currently running is "cur", not "next",
682  * because when we return from switchrame_switch on the
683  * same stack, we're back to the thread that
684  * switchframe_switch call switched away from, which is
685  * "cur".
686  *
687  * - "cur" is _not_ the thread that just *called*
688  * switchframe_switch.
689  *
690  * - If newstate is S_ZOMB we never get back here in that
691  * context at all.
692  *
693  * - If the thread just chosen to run ("next") was a new
694  * thread, we don't get to this code again until
695  * *another* context switch happens, because when new
696  * threads return from switchframe_switch they teleport
697  * to thread_startup.
698  *
699  * - At this point the thread whose stack we're now on may
700  * have been migrated to another cpu since it last ran.
701  *
702  * The above is inherently confusing and will probably take a
703  * while to get used to.
704  *
705  * However, the important part is that code placed here, after
706  * the call to switchframe_switch, does not necessarily run on
707  * every context switch. Thus any such code must be either
708  * skippable on some switches or also called from
709  * thread_startup.
710  */
711 
712 
713  /* Clear the wait channel and set the thread state. */
714  cur->t_wchan_name = NULL;
715  cur->t_state = S_RUN;
716 
717  /* Unlock the run queue. */

830 /*
831  * Thread migration.
832  *
833  * This is also called periodically from hardclock(). If the current
834  * CPU is busy and other CPUs are idle, or less busy, it should move
835  * threads across to those other other CPUs.
836  *
837  * Migrating threads isn't free because of cache affinity; a thread's
838  * working cache set will end up having to be moved to the other CPU,
839  * which is fairly slow. The tradeoff between this performance loss
840  * and the performance loss due to underutilization of some CPUs is
841  * something that needs to be tuned and probably is workload-specific.
842  *
843  * For here and now, because we know we're running on System/161 and
844  * System/161 does not (yet) model such cache effects, we'll be very
845  * aggressive.
846  */
847 void
848 thread_consider_migration(void)
849 {
850  unsigned my_count, total_count, one_share, to_send;
851  unsigned i, numcpus;
852  struct cpu *c;
853  struct threadlist victims;
854  struct thread *t;
855 
856  my_count = total_count = 0;
857  numcpus = cpuarray_num(&allcpus);
858  for (i=0; i<numcpus; i++) {
859  c = cpuarray_get(&allcpus, i);
860  spinlock_acquire(&c->c_runqueue_lock);
861  total_count += c->c_runqueue.tl_count;
862  if (c == curcpu->c_self) {
863  my_count = c->c_runqueue.tl_count;
864  }
865  spinlock_release(&c->c_runqueue_lock);
866  }
867 
868  one_share = DIVROUNDUP(total_count, numcpus);
869  if (my_count < one_share) {
870  return;
871  }
872 
873  to_send = my_count - one_share;
874  threadlist_init(&victims);
875  spinlock_acquire(&curcpu->c_runqueue_lock);
876  for (i=0; i<to_send; i++) {
877  t = threadlist_remtail(&curcpu->c_runqueue);
878  threadlist_addhead(&victims, t);
879  }
880  spinlock_release(&curcpu->c_runqueue_lock);
881 
882  for (i=0; i < numcpus && to_send > 0; i++) {
883  c = cpuarray_get(&allcpus, i);
884  if (c == curcpu->c_self) {
885  continue;
886  }
887  spinlock_acquire(&c->c_runqueue_lock);
888  while (c->c_runqueue.tl_count < one_share && to_send > 0) {
889  t = threadlist_remhead(&victims);
890  /*
891  * Ordinarily, curthread will not appear on
892  * the run queue. However, it can under the
893  * following circumstances:
894  * - it went to sleep;
895  * - the processor became idle, so it
896  * remained curthread;
897  * - it was reawakened, so it was put on the
898  * run queue;
899  * - and the processor hasn't fully unidled
900  * yet, so all these things are still true.
901  *
902  * If the timer interrupt happens at (almost)
903  * exactly the proper moment, we can come here
904  * while things are in this state and see
905  * curthread. However, *migrating* curthread
906  * can cause bad things to happen (Exercise:
907  * Why? And what?) so shuffle it to the end of
908  * the list and decrement to_send in order to
909  * skip it. Then it goes back on our own run
910  * queue below.
911  */
912  if (t == curthread) {
913  threadlist_addtail(&victims, t);
914  to_send--;
915  continue;
916  }
917 
918  t->t_cpu = c;
919  threadlist_addtail(&c->c_runqueue, t);
920  DEBUG(DB_THREADS,
921  "Migrated thread %s: cpu %u -> %u",
922  t->t_name, curcpu->c_number, c->c_number);
923  to_send--;
924  if (c->c_isidle) {
925  /*
926  * Other processor is idle; send
927  * interrupt to make sure it unidles.
928  */
929  ipi_send(c, IPI_UNIDLE);
930  }
931  }
932  spinlock_release(&c->c_runqueue_lock);
933  }
934 
935  /*
936  * Because the code above isn't atomic, the thread counts may have
937  * changed while we were working and we may end up with leftovers.
938  * Don't panic; just put them back on our own run queue.
939  */
940  if (!threadlist_isempty(&victims)) {
941  spinlock_acquire(&curcpu->c_runqueue_lock);
942  while ((t = threadlist_remhead(&victims)) != NULL) {
943  threadlist_addtail(&curcpu->c_runqueue, t);
944  }
945  spinlock_release(&curcpu->c_runqueue_lock);
946  }
947 
948  KASSERT(threadlist_isempty(&victims));
949  threadlist_cleanup(&victims);
950 }
