55 struct ibstate {

58  uint32_t fileblocks; /* file size in blocks (constant) */

60  unsigned pasteofcount; /* number of blocks found past eof */
61  blockusage_t usagetype; /* how to call freemap_blockinuse() */

64 /*
65  * Traverse an indirect block, recording blocks that are in use,
66  * dropping any entries that are past EOF, and clearing any entries
67  * that point outside the volume.
68  *
69  * XXX: this should be extended to be able to recover from crosslinked
70  * blocks. Currently it just complains in freemap.c and sets
71  * EXIT_UNRECOV.
72  *
73  * The traversal is recursive; the state is maintained in IBS (as
74  * described above). IENTRY is a pointer to the entry in the parent
75  * indirect block (or the inode) that names the block we're currently
76  * scanning. IECHANGEDP should be set to 1 if *IENTRY is changed.
77  * INDIRECTION is the indirection level of this block (1, 2, or 3).
78  */
79 static
80 void
81 check_indirect_block(struct ibstate *ibs, uint32_t *ientry, int *iechangedp,
82  int indirection)
83 {
84  uint32_t entries[SFS_DBPERIDB];
85  uint32_t i, ct;
86  uint32_t coveredblocks;
87  int localchanged = 0;
88  int j;
89 
90  if (*ientry > 0 && *ientry < ibs->volblocks) {
91  sfs_readindirect(*ientry, entries);
92  freemap_blockinuse(*ientry, B_IBLOCK, ibs->ino);
93  }
94  else {
95  if (*ientry >= ibs->volblocks) {
96  setbadness(EXIT_RECOV);
97  warnx("Inode %lu: indirect block pointer (level %d) "
98  "for block %lu outside of volume: %lu "
99  "(cleared)\n",
100  (unsigned long)ibs->ino, indirection,
101  (unsigned long)ibs->curfileblock,
102  (unsigned long)*ientry);
103  *ientry = 0;
104  *iechangedp = 1;
105  }
106  coveredblocks = 1;
107  for (j=0; j<indirection; j++) {
108  coveredblocks *= SFS_DBPERIDB;
109  }
110  ibs->curfileblock += coveredblocks;
111  return;
112  }
113 
114  if (indirection > 1) {
115  for (i=0; i<SFS_DBPERIDB; i++) {
116  check_indirect_block(ibs, &entries[i], &localchanged,
117  indirection-1);
118  }
119  }
120  else {
121  assert(indirection==1);
122 
123  for (i=0; i<SFS_DBPERIDB; i++) {
124  if (entries[i] >= ibs->volblocks) {
125  setbadness(EXIT_RECOV);
126  warnx("Inode %lu: direct block pointer for "
127  "block %lu outside of volume: %lu "
128  "(cleared)\n",
129  (unsigned long)ibs->ino,
130  (unsigned long)ibs->curfileblock,
131  (unsigned long)entries[i]);
132  entries[i] = 0;
133  localchanged = 1;
134  }
135  else if (entries[i] != 0) {
136  if (ibs->curfileblock < ibs->fileblocks) {
137  freemap_blockinuse(entries[i],
138  ibs->usagetype,
139  ibs->ino);
140  }
141  else {
142  setbadness(EXIT_RECOV);
143  ibs->pasteofcount++;
144  freemap_blockfree(entries[i]);
145  entries[i] = 0;
146  localchanged = 1;
147  }
148  }
149  ibs->curfileblock++;
150  }
151  }
152 
153  ct=0;
154  for (i=ct=0; i<SFS_DBPERIDB; i++) {
155  if (entries[i]!=0) ct++;
156  }
157  if (ct==0) {
158  if (*ientry != 0) {
159  setbadness(EXIT_RECOV);
160  /* this is not necessarily correct */
161  /*ibs->pasteofcount++;*/
162  *iechangedp = 1;
163  freemap_blockfree(*ientry);
164  *ientry = 0;
165  }
166  }
167  else {
168  assert(*ientry != 0);
169  if (localchanged) {
170  sfs_writeindirect(*ientry, entries);
171  }
172  }
173 }

175 /*
176  * Check the blocks belonging to inode INO, whose inode has already
177  * been loaded into SFI. ISDIR is a shortcut telling us if the inode
178  * is a directory.
179  *
180  * Returns nonzero if SFI has been modified and needs to be written
181  * back.
182  */
183 static
184 int
185 check_inode_blocks(uint32_t ino, struct sfs_dinode *sfi, int isdir)
186 {
187  struct ibstate ibs;
188  uint32_t size, datablock;
189  int changed;
190  int i;
191 
192  size = SFS_ROUNDUP(sfi->sfi_size, SFS_BLOCKSIZE);
193 
194  ibs.ino = ino;
195  /*ibs.curfileblock = 0;*/
196  ibs.fileblocks = size/SFS_BLOCKSIZE;
197  ibs.volblocks = sb_totalblocks();
198  ibs.pasteofcount = 0;
199  ibs.usagetype = isdir ? B_DIRDATA : B_DATA;
200 
201  changed = 0;
202 
203  for (ibs.curfileblock=0; ibs.curfileblock<NUM_D; ibs.curfileblock++) {
204  datablock = GET_D(sfi, ibs.curfileblock);
205  if (datablock >= ibs.volblocks) {
206  setbadness(EXIT_RECOV);
207  warnx("Inode %lu: direct block pointer for "
208  "block %lu outside of volume: %lu "
209  "(cleared)\n",
210  (unsigned long)ibs.ino,
211  (unsigned long)ibs.curfileblock,
212  (unsigned long)datablock);
213  SET_D(sfi, ibs.curfileblock) = 0;
214  changed = 1;
215  }
216  else if (datablock > 0) {
217  if (ibs.curfileblock < ibs.fileblocks) {
218  freemap_blockinuse(datablock, ibs.usagetype,
219  ibs.ino);
220  }
221  else {
222  setbadness(EXIT_RECOV);
223  ibs.pasteofcount++;
224  changed = 1;
225  freemap_blockfree(datablock);
226  SET_D(sfi, ibs.curfileblock) = 0;
227  }
228  }
229  }
230 
231  for (i=0; i<NUM_I; i++) {
232  check_indirect_block(&ibs, &SET_I(sfi, i), &changed, 1);
233  }
234  for (i=0; i<NUM_II; i++) {
235  check_indirect_block(&ibs, &SET_II(sfi, i), &changed, 2);
236  }
237  for (i=0; i<NUM_III; i++) {
238  check_indirect_block(&ibs, &SET_III(sfi, i), &changed, 3);
239  }
240 
241  if (ibs.pasteofcount > 0) {
242  warnx("Inode %lu: %u blocks after EOF (freed)",
243  (unsigned long) ibs.ino, ibs.pasteofcount);
244  setbadness(EXIT_RECOV);
245  }
246 
247  return changed;
248 }

250 /*
251  * Do the pass1 inode-level checks on inode INO, which has already
252  * been loaded into SFI. Note that sfi_type has already been
253  * validated.
254  *
255  * Returns nonzero if SFI has been modified and needs to be written
256  * back.
257  */
258 static
259 int
260 pass1_inode(uint32_t ino, struct sfs_dinode *sfi, int alreadychanged)
261 {
262  int changed = alreadychanged;
263  int isdir = sfi->sfi_type == SFS_TYPE_DIR;
264 
265  if (inode_add(ino, sfi->sfi_type)) {
266  /* Already been here. */
267  assert(changed == 0);
268  return 1;
269  }
270 
271  freemap_blockinuse(ino, B_INODE, ino);
272 
273  if (checkzeroed(sfi->sfi_waste, sizeof(sfi->sfi_waste))) {
274  warnx("Inode %lu: sfi_waste section not zeroed (fixed)",
275  (unsigned long) ino);
276  setbadness(EXIT_RECOV);
277  changed = 1;
278  }
279 
280  if (check_inode_blocks(ino, sfi, isdir)) {
281  changed = 1;
282  }
283 
284  if (changed) {
285  sfs_writeinode(ino, sfi);
286  }
287  return 0;
288 }

290 /*
291  * Check the directory entry in SFD. INDEX is its offset, and PATH is
292  * its name; these are used for printing messages.
293  */
294 static
295 int
296 pass1_direntry(const char *path, uint32_t index, struct sfs_direntry *sfd)
297 {
298  int dchanged = 0;
299  uint32_t nblocks;
300 
301  nblocks = sb_totalblocks();
302 
303  if (sfd->sfd_ino == SFS_NOINO) {
304  if (sfd->sfd_name[0] != 0) {
305  setbadness(EXIT_RECOV);
306  warnx("Directory %s entry %lu has name but no file",
307  path, (unsigned long) index);
308  sfd->sfd_name[0] = 0;
309  dchanged = 1;
310  }
311  }
312  else if (sfd->sfd_ino >= nblocks) {
313  setbadness(EXIT_RECOV);
314  warnx("Directory %s entry %lu has out of range "
315  "inode (cleared)",
316  path, (unsigned long) index);
317  sfd->sfd_ino = SFS_NOINO;
318  sfd->sfd_name[0] = 0;
319  dchanged = 1;
320  }
321  else {
322  if (sfd->sfd_name[0] == 0) {
323  /* XXX: what happens if FSCK.n.m already exists? */
324  snprintf(sfd->sfd_name, sizeof(sfd->sfd_name),
325  "FSCK.%lu.%lu",
326  (unsigned long) sfd->sfd_ino,
327  (unsigned long) uniqueid());
328  setbadness(EXIT_RECOV);
329  warnx("Directory %s entry %lu has file but "
330  "no name (fixed: %s)",
331  path, (unsigned long) index,
332  sfd->sfd_name);
333  dchanged = 1;
334  }
335  if (checknullstring(sfd->sfd_name, sizeof(sfd->sfd_name))) {
336  setbadness(EXIT_RECOV);
337  warnx("Directory %s entry %lu not "
338  "null-terminated (fixed)",
339  path, (unsigned long) index);
340  dchanged = 1;
341  }
342  if (checkbadstring(sfd->sfd_name)) {
343  setbadness(EXIT_RECOV);
344  warnx("Directory %s entry %lu contains invalid "
345  "characters (fixed)",
346  path, (unsigned long) index);
347  dchanged = 1;
348  }
349  }
350  return dchanged;
351 }

353 /*
354  * Check a directory. INO is the inode number; PATHSOFAR is the path
355  * to this directory. This traverses the volume directory tree
356  * recursively.
357  */
358 static
359 void
360 pass1_dir(uint32_t ino, const char *pathsofar)
361 {
362  struct sfs_dinode sfi;
363  struct sfs_direntry *direntries;
364  uint32_t ndirentries, i;
365  int ichanged=0, dchanged=0;
366 
367  sfs_readinode(ino, &sfi);
368 
369  if (sfi.sfi_size % sizeof(struct sfs_direntry) != 0) {
370  setbadness(EXIT_RECOV);
371  warnx("Directory %s has illegal size %lu (fixed)",
372  pathsofar, (unsigned long) sfi.sfi_size);
373  sfi.sfi_size = SFS_ROUNDUP(sfi.sfi_size,
374  sizeof(struct sfs_direntry));
375  ichanged = 1;
376  }
377  count_dirs++;
378 
379  if (pass1_inode(ino, &sfi, ichanged)) {
380  /* been here before; crosslinked dir, sort it out in pass 2 */
381  return;
382  }
383 
384  ndirentries = sfi.sfi_size/sizeof(struct sfs_direntry);
385  direntries = domalloc(sfi.sfi_size);
386 
387  sfs_readdir(&sfi, direntries, ndirentries);
388 
389  for (i=0; i<ndirentries; i++) {
390  if (pass1_direntry(pathsofar, i, &direntries[i])) {
391  dchanged = 1;
392  }
393  }
394 
395  for (i=0; i<ndirentries; i++) {
396  if (direntries[i].sfd_ino == SFS_NOINO) {
397  /* nothing */
398  }
399  else if (!strcmp(direntries[i].sfd_name, ".")) {
400  /* nothing */
401  }
402  else if (!strcmp(direntries[i].sfd_name, "..")) {
403  /* nothing */
404  }
405  else {
406  char path[strlen(pathsofar)+SFS_NAMELEN+1];
407  struct sfs_dinode subsfi;
408  uint32_t subino;
409 
410  subino = direntries[i].sfd_ino;
411  sfs_readinode(subino, &subsfi);
412  snprintf(path, sizeof(path), "%s/%s",
413  pathsofar, direntries[i].sfd_name);
414 
415  switch (subsfi.sfi_type) {
416  case SFS_TYPE_FILE:
417  if (pass1_inode(subino, &subsfi, 0)) {
418  /* been here before */
419  break;
420  }
421  count_files++;
422  break;
423  case SFS_TYPE_DIR:
424  pass1_dir(subino, path);
425  break;
426  default:
427  setbadness(EXIT_RECOV);
428  warnx("Object %s: Invalid inode type %u "
429  "(removed)", path, subsfi.sfi_type);
430  direntries[i].sfd_ino = SFS_NOINO;
431  direntries[i].sfd_name[0] = 0;
432  dchanged = 1;
433  break;
434  }
435  }
436  }
437 
438  if (dchanged) {
439  sfs_writedir(&sfi, direntries, ndirentries);
440  }
441 
442  free(direntries);
443 }

445 /*
446  * Check the root directory, and implicitly everything under it.
447  */
448 static
449 void
450 pass1_rootdir(void)
451 {
452  struct sfs_dinode sfi;
453  char path[SFS_VOLNAME_SIZE + 2];
454 
455  sfs_readinode(SFS_ROOTDIR_INO, &sfi);
456 
457  switch (sfi.sfi_type) {
458  case SFS_TYPE_DIR:
459  break;
460  case SFS_TYPE_FILE:
461  warnx("Root directory inode is a regular file (fixed)");
462  goto fix;
463  default:
464  warnx("Root directory inode has invalid type %lu (fixed)",
465  (unsigned long) sfi.sfi_type);
466  fix:
467  setbadness(EXIT_RECOV);
468  sfi.sfi_type = SFS_TYPE_DIR;
469  sfs_writeinode(SFS_ROOTDIR_INO, &sfi);
470  break;
471  }
472 
473  snprintf(path, sizeof(path), "%s:", sb_volname());
474  pass1_dir(SFS_ROOTDIR_INO, path);
475 }
