62 {
63  unsigned a = *(const unsigned *)av;
64  unsigned b = *(const unsigned *)bv;
65  unsigned ax = (a & 0xffff0000) >> 16;
66  unsigned ay = a & 0xffff;
67  unsigned bx = (b & 0xffff0000) >> 16;
68  unsigned by = b & 0xffff;
69 
70  if (ax < bx) {
71  return 1;

73  if (ax > bx) {
74  return -1;
75  }
76  if (ay < by) {
77  return -1;
78  }
79  if (ay > by) {

81  }

138 /*
139  * grind - memory-bound task
140  *
141  * Note that this won't work until you have a VM system.
142  */
143 void
144 grind(unsigned groupid, unsigned id)
145 {
146  unsigned *p;
147  unsigned i, n, s;
148 
149  (void)groupid;
150 
151  waitstart();
152 
153  /* each grind task uses 768K */
154  n = (768*1024) / sizeof(*p);
155  p = malloc(n * sizeof(*p));
156  if (p == NULL) {
157  if (errno == ENOSYS) {
158  /*
159  * If we don't have sbrk, just bail out with
160  * "success" instead of failing the whole
161  * workload.
162  */
163  errx(0, "grind: sbrk/malloc not implemented");
164  }
165  err(1, "malloc");
166  }
167 
168  /* First, get some random integers. */
169  warnx("grind %u: seeding", id);
170  srandom(1753);
171  for (i=0; i<n; i++) {
172  p[i] = random();
173  }
174 
175  /* Now sort them. */
176  warnx("grind %u: sorting", id);
177  qsort(p, n, sizeof(p[0]), uintcmp);
178 
179  /* Sort by a different comparison. */
180  warnx("grind %u: sorting alternately", id);
181  qsort(p, n, sizeof(p[0]), altcmp);
182 
183  /* Take the sum. */
184  warnx("grind %u: summing", id);
185  s = sum(p, n);
186  warnx("grind %u: sum is %u (should be %u)", id, s, RIGHT);
187  if (s != RIGHT) {
188  errx(1, "grind %u FAILED", id);
189  }
190 
191  /* Take first differences. */
192  warnx("grind %u: first differences", id);
193  diffs(p, n);
194 
195  /* Sort. */
196  warnx("grind %u: sorting", id);
197  qsort(p, n, sizeof(p[0]), uintcmp);
198 
199  warnx("grind %u: summing", id);
200  s = sum(p, n);
201  warnx("grind %u: sum is %u (should be 0)", id, s);
202  if (s != 0) {
203  errx(1, "grind %u FAILED", id);
204  }
205 }
