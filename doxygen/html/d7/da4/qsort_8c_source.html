35  * qsort() for OS/161, where it isn't in libc.
36  */
37 void
38 qsort(void *vdata, unsigned num, size_t size,
39  int (*f)(const void *, const void *))
40 {
41  unsigned pivot, head, tail;
42  char *data = vdata;
43  char tmp[size];
44 
45 #define COMPARE(aa, bb) \
46  ((aa) == (bb) ? 0 : f(data + (aa) * size, data + (bb) * size))
47 #define EXCHANGE(aa, bb) \
48  memcpy(tmp, data + (aa) * size, size); \
49  memcpy(data + (aa) * size, data + (bb) * size, size); \
50  memcpy(data + (bb) * size, tmp, size)
51 
52 
53  if (num <= 1) {
54  return;
55  }
56  if (num == 2) {
57  if (COMPARE(0, 1) > 0) {
58  EXCHANGE(0, 1);
59  return;
60  }
61  }
62 
63  /*
64  * 1. Pick a pivot value. For simplicity, always use the
65  * middle of the array.
66  */
67  pivot = num / 2;
68 
69  /*
70  * 2. Shift all values less than or equal to the pivot value
71  * to the front of the array.
72  */
73  head = 0;
74  tail = num - 1;
75 
76  while (head < tail) {
77  if (COMPARE(head, pivot) <= 0) {
78  head++;
79  }
80  else if (COMPARE(tail, pivot) > 0) {
81  tail--;
82  }
83  else {
84  EXCHANGE(head, tail);
85  if (pivot == head) {
86  pivot = tail;
87  }
88  else if (pivot == tail) {
89  pivot = head;
90  }
91  head++;
92  tail--;
93  }
94  }
95 
96  /*
97  * 3. If there's an even number of elements and we swapped the
98  * last two, the head and tail indexes will cross. In this
99  * case the first entry on the tail side is tail+1. If there's
100  * an odd number of elements, we stop with head == tail, and
101  * the first entry on the tail side is this value (hence,
102  * tail) if it's is greater than the pivot value, and the next
103  * element (hence, tail+1) if it's less than or equal to the
104  * pivot value.
105  *
106  * Henceforth use "tail" to hold the index of the first entry
107  * of the back portion of the array.
108  */
109  if (head > tail || COMPARE(head, pivot) <= 0) {
110  tail++;
111  }
112 
113  /*
114  * 4. If we got a bad pivot that gave us only one partition,
115  * because of the order of the advances in the loop above it
116  * will always put everything in the front portion of the
117  * array (so tail == num). This happens if we picked the
118  * largest value. Move the pivot to the end, if necessary, lop
119  * off all values equal to it, and recurse on the rest. (If
120  * there is no rest, the array is already sorted and we're
121  * done.)
122  */
123  if (tail == num) {
124  if (pivot < num - 1) {
125  if (COMPARE(pivot, num - 1) > 0) {
126  EXCHANGE(pivot, num - 1);
127  }
128  }
129  tail = num - 1;
130  while (tail > 0 && COMPARE(tail - 1, tail) == 0) {
131  tail--;
132  }
133  if (tail > 0) {
134  qsort(vdata, tail, size, f);
135  }
136  return;
137  }
138  assert(tail > 0 && tail < num);
139 
140  /*
141  * 5. Recurse on each subpart of the array.
142  */
143  qsort(vdata, tail, size, f);
144  qsort((char *)vdata + tail * size, num - tail, size, f);
145 }
