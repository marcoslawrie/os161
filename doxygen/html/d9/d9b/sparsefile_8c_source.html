30 /*
31  * Create a sparse file by writing one byte to the end of it.
32  *
33  * Should work on emufs (emu0:) once the basic system calls are done,
34  * and should work on SFS when the file system assignment is
35  * done. Sufficiently small files should work on SFS even before that
36  * assignment.
37  */

45 int
46 main(int argc, char *argv[])
47 {
48  const char *filename;
49  int size;
50  int fd;
51  int r;
52  char byte;
53 
54  if (argc != 3) {
55  errx(1, "Usage: sparsefile <filename> <size>");
56  }
57 
58  filename = argv[1];
59  size = atoi(argv[2]);
60  byte = '\n';
61 
62  if (size == 0) {
63  err(1, "Sparse files of length zero are not meaningful");
64  }
65 
66  printf("Creating a sparse file of size %d\n", size);
67 
68  fd = open(filename, O_WRONLY|O_CREAT|O_TRUNC);
69  if (fd < 0) {
70  err(1, "%s: create", filename);
71  }
72 
73  if (lseek(fd, size-1, SEEK_SET) == -1) {
74  err(1, "%s: lseek", filename);
75  }
76  r = write(fd, &byte, 1);
77  if (r < 0) {
78  err(1, "%s: write", filename);
79  }
80  else if (r != 1) {
81  errx(1, "%s: write: Unexpected result count %d", filename, r);
82  }
83 
84  close(fd);
85 
86  return 0;
87 }
