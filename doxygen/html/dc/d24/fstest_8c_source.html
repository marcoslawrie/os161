74 /*
75  * Vary each line of the test file in a way that's predictable but
76  * unlikely to mask bugs in the filesystem.
77  */
78 static
79 void
80 rotate(char *str, int amt)
81 {
82  int i, ch;
83 
84  amt = (amt+2600)%26;
85  KASSERT(amt>=0);
86 
87  for (i=0; str[i]; i++) {
88  ch = str[i];
89  if (ch>='A' && ch<='Z') {
90  ch = ch - 'A';
91  ch += amt;
92  ch %= 26;
93  ch = ch + 'A';
94  KASSERT(ch>='A' && ch<='Z');
95  }
96  str[i] = ch;
97  }
98 }

133 static
134 int
135 fstest_write(const char *fs, const char *namesuffix,
136  int stridesize, int stridepos)
137 {
138  struct vnode *vn;
139  int err;
140  int i;
141  size_t shouldbytes=0;
142  size_t bytes=0;
143  off_t pos=0;
144  char name[32];
145  char buf[32];
146  struct iovec iov;
147  struct uio ku;
148  int flags;
149 
150  KASSERT(sizeof(buf) > strlen(SLOGAN));
151 
152  MAKENAME();
153 
154  flags = O_WRONLY|O_CREAT;
155  if (stridesize == 1) {
156  flags |= O_TRUNC;
157  }
158 
159  /* vfs_open destroys the string it's passed */
160  strcpy(buf, name);
161  err = vfs_open(buf, flags, 0664, &vn);
162  if (err) {
163  kprintf("Could not open %s for write: %s\n",
164  name, strerror(err));
165  return -1;
166  }
167 
168  for (i=0; i<NCHUNKS; i++) {
169  if (i % stridesize != stridepos) {
170  pos += strlen(SLOGAN);
171  continue;
172  }
173  strcpy(buf, SLOGAN);
174  rotate(buf, i);
175  uio_kinit(&iov, &ku, buf, strlen(SLOGAN), pos, UIO_WRITE);
176  err = VOP_WRITE(vn, &ku);
177  if (err) {
178  kprintf("%s: Write error: %s\n", name, strerror(err));
179  vfs_close(vn);
180  vfs_remove(name);
181  return -1;
182  }
183 
184  if (ku.uio_resid > 0) {
185  kprintf("%s: Short write: %lu bytes left over\n",
186  name, (unsigned long) ku.uio_resid);
187  vfs_close(vn);
188  vfs_remove(name);
189  return -1;
190  }
191 
192  bytes += (ku.uio_offset - pos);
193  shouldbytes += strlen(SLOGAN);
194  pos = ku.uio_offset;
195  }
196 
197  vfs_close(vn);
198 
199  if (bytes != shouldbytes) {
200  kprintf("%s: %lu bytes written, should have been %lu!\n",
201  name, (unsigned long) bytes,
202  (unsigned long) (NCHUNKS*strlen(SLOGAN)));
203  vfs_remove(name);
204  return -1;
205  }
206  kprintf("%s: %lu bytes written\n", name, (unsigned long) bytes);
207 
208  return 0;
209 }

212 int
213 fstest_read(const char *fs, const char *namesuffix)
214 {
215  struct vnode *vn;
216  int err;
217  int i;
218  size_t bytes=0;
219  char name[32];
220  char buf[32];
221  struct iovec iov;
222  struct uio ku;
223 
224  MAKENAME();
225 
226  /* vfs_open destroys the string it's passed */
227  strcpy(buf, name);
228  err = vfs_open(buf, O_RDONLY, 0664, &vn);
229  if (err) {
230  kprintf("Could not open test file for read: %s\n",
231  strerror(err));
232  return -1;
233  }
234 
235  for (i=0; i<NCHUNKS; i++) {
236  uio_kinit(&iov, &ku, buf, strlen(SLOGAN), bytes, UIO_READ);
237  err = VOP_READ(vn, &ku);
238  if (err) {
239  kprintf("%s: Read error: %s\n", name, strerror(err));
240  vfs_close(vn);
241  return -1;
242  }
243 
244  if (ku.uio_resid > 0) {
245  kprintf("%s: Short read: %lu bytes left over\n", name,
246  (unsigned long) ku.uio_resid);
247  vfs_close(vn);
248  return -1;
249  }
250  buf[strlen(SLOGAN)] = 0;
251  rotate(buf, -i);
252  if (strcmp(buf, SLOGAN)) {
253  kprintf("%s: Test failed: line %d mismatched: %s\n",
254  name, i+1, buf);
255  vfs_close(vn);
256  return -1;
257  }
258 
259  bytes = ku.uio_offset;
260  }
261 
262  vfs_close(vn);
263 
264  if (bytes != NCHUNKS*strlen(SLOGAN)) {
265  kprintf("%s: %lu bytes read, should have been %lu!\n",
266  name, (unsigned long) bytes,
267  (unsigned long) (NCHUNKS*strlen(SLOGAN)));
268  return -1;
269  }
270  kprintf("%s: %lu bytes read\n", name, (unsigned long) bytes);
271  return 0;
272 }
