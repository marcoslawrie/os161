39 /* Lowest revision we support */
40 #define LOW_VERSION 1
41 
42 struct ltimer_softc *
43 attach_ltimer_to_lamebus(int ltimerno, struct lamebus_softc *sc)
44 {
45  struct ltimer_softc *lt;
46  int slot = lamebus_probe(sc, LB_VENDOR_CS161, LBCS161_TIMER,
47  LOW_VERSION, NULL);
48  if (slot < 0) {
49  /* No ltimer (or no additional ltimer) found */
50  return NULL;
51  }
52 
53  lt = kmalloc(sizeof(struct ltimer_softc));
54  if (lt==NULL) {
55  /* out of memory */
56  return NULL;
57  }
58 
59  (void)ltimerno; // unused
60 
61  /* Record what bus it's on */
62  lt->lt_bus = sc;
63  lt->lt_buspos = slot;
64 
65  /* Mark the slot in use and hook that slot's interrupt */
66  lamebus_mark(sc, slot);
67  lamebus_attach_interrupt(sc, slot, lt, ltimer_irq);
68 
69  return lt;
70 }
