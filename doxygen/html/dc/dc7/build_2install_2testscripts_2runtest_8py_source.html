2 # Usage:
3 # import runtest
4 # runtest.run(testcommands, outputfile,
5 # menuprompt=None, default "OS/161 kernel [? for menu]: "
6 # shellprompt=None, default "OS/161$ "
7 # conf=None, default is sys161 default behavior
8 # ram=None, default is per sys161 config
9 # cpus=None, default is per sys161 config
10 # doom=None, default is no doom counter
11 # progress=30, default is 30 seconds
12 # timeout=300, default is 300 seconds
13 # kernel=None) default is "kernel"
14 #
15 # Returns None on success or a (string) message if something apparently
16 # went wrong in the middle. (XXX: should it throw exceptions instead?)
17 #
18 # * The testcommands argument is a string containing a list of commands
19 # separated by semicolons. These can be either kernel menu commands
20 # or shell commands; the command 's' is recognized for switching from
21 # the menu to the shell and 'exit' for switching back to the menu.
22 # (This affects waiting for prompts - running the shell via 'p' or
23 # crashing out of the shell will confuse things.)
24 #
25 # The command 'q' from the menu is also recognized as causing a
26 # shutdown. This will be done automatically after everything else if
27 # not issued explicitly.
28 #
29 # The following commands are interpreted as macros:
30 # DOMOUNT expands to "mount sfs lhd1:; cd lhd1:"
31 # DOUNMOUNT expands to "cd /; unmount lhd1:"
32 # WAIT sleeps 3 seconds and just presses return
33 #
34 # * The outputfile argument should be a python file (e.g. sys.stdout)
35 # and receives a copy of the System/161 output.
36 #
37 # * The menuprompt and shellprompt arguments can be used to change the
38 # menu and shell prompt strings looked for. For the moment these can
39 # only be fixed strings, not regular expressions. (This is probably
40 # easy to improve, but I ran into some mysterious problems when I
41 # tried, so YMMV.) By default if you pass None prompt strings matching
42 # what OS/161 issues by default are used.
43 #
44 # * The conf argument can be used to supply an alternate sys161.conf
45 # file. If None is given (the default), sys161 will use its default
46 # config file.
47 #
48 # * The ram and cpus arguments can be used to override the RAM size
49 # and number-of-cpus settings in the sys161 config file. The number of
50 # cpus must be an integer, but any RAM size specification understood
51 # by sys161 can be used. Note: this feature requires System/161 2.0.5
52 # or higher.
53 #
54 # * The doom argument can be used to set the doom counter. If None is
55 # given (the default) the doom counter is not engaged.
56 #
57 # * The progress and timeout arguments can be used to set the timeouts
58 # for System/161 progress monitoring and pexpect-level global timeout,
59 # respectively. The defaults (somewhat arbitraily chosen) are 30 and
60 # 300 seconds. Passing progress=None disables progress monitoring; this
61 # is necessary for nontrivial tests that run within the kernel, as
62 # progress monitoring measures userland progress. Passing timeout=None
63 # probably either disables the global timeout or makes pexpect crash;
64 # I haven't tested it. I don't recommend trying: it is your defense
65 # against test runs hanging forever.
66 #
67 # Note that no-debugger unattended mode (sys161 -X) is always used.
68 # The purpose of this script is specifically to support unattended
69 # test runs...
70 #
71 # Depends on pexpect, which you may need to install specifically
72 # depending on your OS.
73 #

91 def getprompt(proc, prompt):
92  which = proc.expect_exact([
93  prompt,
94  "panic: ", # panic message
95  "sys161: No progress in ", # sys161 deadman print
96  "sys161: Elapsed ", # sys161 shutdown print
97  pexpect.EOF,
98  pexpect.TIMEOUT
99  ])
100  if which == 0:
101  # got the prompt
102  return None
103  if which == 1:
104  proc.expect_exact([pexpect.EOF, pexpect.TIMEOUT])
105  return "panic"
106  if which == 2:
107  proc.expect_exact([pexpect.EOF, pexpect.TIMEOUT])
108  return "progress timeout"
109  if which == 3:
110  proc.expect_exact([pexpect.EOF, pexpect.TIMEOUT])
111  return "unexpected shutdown"
112  if which == 4:
113  return "unexpected end of input"
114  if which == 5:
