122 /*
123  * Indirect block bmap: in indirect block IBLOCK, read the entry at
124  * block OFFSET from the first file block mapped by this indirect
125  * block.
126  *
127  * ENTRYSIZE is how many blocks each entry describes; for a
128  * singly-indirect block this is 1. For a multiply-indirect block,
129  * it is more than 1; in this case recurse.
130  */
131 static
132 uint32_t
133 ibmap(uint32_t iblock, uint32_t offset, uint32_t entrysize)
134 {
135  uint32_t entries[SFS_DBPERIDB];
136 
137  if (iblock == 0) {
138  return 0;
139  }
140 
141  diskread(entries, iblock);
142  swapindir(entries);
143 
144  if (entrysize > 1) {
145  uint32_t index = offset / entrysize;
146  offset %= entrysize;
147  return ibmap(entries[index], offset, entrysize/SFS_DBPERIDB);
148  }
149  else {
150  assert(offset < SFS_DBPERIDB);
151  return entries[offset];
152  }
153 }

155 /*
156  * bmap() for SFS.
157  *
158  * Given an inode and a file block, returns a disk block.
159  */
160 static
161 uint32_t
162 bmap(const struct sfs_dinode *sfi, uint32_t fileblock)
163 {
164  uint32_t iblock, offset;
165 
166  if (fileblock < INOMAX_D) {
167  return GET_D(sfi, fileblock);
168  }
169  else if (fileblock < INOMAX_I) {
170  iblock = (fileblock - INOMAX_D) / RANGE_I;
171  offset = (fileblock - INOMAX_D) % RANGE_I;
172  return ibmap(GET_I(sfi, iblock), offset, RANGE_D);
173  }
174  else if (fileblock < INOMAX_II) {
175  iblock = (fileblock - INOMAX_I) / RANGE_II;
176  offset = (fileblock - INOMAX_I) % RANGE_II;
177  return ibmap(GET_II(sfi, iblock), offset, RANGE_I);
178  }
179  else if (fileblock < INOMAX_III) {
180  iblock = (fileblock - INOMAX_II) / RANGE_III;
181  offset = (fileblock - INOMAX_II) % RANGE_III;
182  return ibmap(GET_III(sfi, iblock), offset, RANGE_II);
183  }
184  return 0;
185 }

270 /*
271  * Read the directory block at DISKBLOCK into D.
272  */
273 static
274 void
275 sfs_readdirblock(struct sfs_direntry *d, uint32_t diskblock)
276 {
277  const unsigned atonce = SFS_BLOCKSIZE/sizeof(struct sfs_direntry);
278  unsigned j;
279 
280  if (diskblock != 0) {
281  diskread(d, diskblock);
282  for (j=0; j<atonce; j++) {
283  swapdir(&d[j]);
284  }
285  }
286  else {
287  warnx("Warning: sparse directory found");
288  bzero(d, SFS_BLOCKSIZE);
289  }
290 }

292 /*
293  * Read in a directory, from the inode SFI, into D, which is a buffer
294  * with ND slots. The caller is assumed to have figured out the right
295  * number of slots.
296  */
297 void
298 sfs_readdir(struct sfs_dinode *sfi, struct sfs_direntry *d, unsigned nd)
299 {
300  const unsigned atonce = SFS_BLOCKSIZE/sizeof(struct sfs_direntry);
301  unsigned nblocks = SFS_ROUNDUP(nd, atonce) / atonce;
302  unsigned i, j;
303  unsigned left, thismany;
304  struct sfs_direntry buffer[atonce];
305  uint32_t diskblock;
306 
307  left = nd;
308  for (i=0; i<nblocks; i++) {
309  diskblock = bmap(sfi, i);
310  if (left < atonce) {
311  thismany = left;
312  sfs_readdirblock(buffer, diskblock);
313  for (j=0; j<thismany; j++) {
314  d[i*atonce + j] = buffer[j];
315  }
316  }
317  else {
318  thismany = atonce;
319  sfs_readdirblock(d + i*atonce, diskblock);
320  }
321  left -= thismany;
322  }
323  assert(left == 0);
324 }

326 /*
327  * Write the directory block D to DISKBLOCK.
328  */
329 static
330 void
331 sfs_writedirblock(struct sfs_direntry *d, uint32_t diskblock)
332 {
333  const unsigned atonce = SFS_BLOCKSIZE/sizeof(struct sfs_direntry);
334  unsigned j, bad;
335 
336  if (diskblock != 0) {
337  for (j=0; j<atonce; j++) {
338  swapdir(&d[j]);
339  }
340  diskwrite(d, diskblock);
341  }
342  else {
343  for (j=bad=0; j<atonce; j++) {
344  if (d[j].sfd_ino != SFS_NOINO ||
345  d[j].sfd_name[0] != 0) {
346  bad = 1;
347  }
348  }
349  if (bad) {
350  warnx("Cannot write to missing block in "
351  "sparse directory (ERROR)");
352  setbadness(EXIT_UNRECOV);
353  }
354  }
355 }

357 /*
358  * Write out a directory, from the inode SFI, using D, which is a
359  * buffer with ND slots. The caller is assumed to have set the inode
360  * size accordingly.
361  */
362 void
363 sfs_writedir(const struct sfs_dinode *sfi, struct sfs_direntry *d, unsigned nd)
364 {
365  const unsigned atonce = SFS_BLOCKSIZE/sizeof(struct sfs_direntry);
366  unsigned nblocks = SFS_ROUNDUP(nd, atonce) / atonce;
367  unsigned i, j;
368  unsigned left, thismany;
369  struct sfs_direntry buffer[atonce];
370  uint32_t diskblock;
371 
372  left = nd;
373  for (i=0; i<nblocks; i++) {
374  diskblock = bmap(sfi, i);
375  if (left < atonce) {
376  thismany = left;
377  for (j=0; j<thismany; j++) {
378  buffer[j] = d[i*atonce + j];
379  }
380  for (; j<atonce; j++) {
381  memset(&buffer[j], 0, sizeof(buffer[j]));
382  }
383  sfs_writedirblock(buffer, diskblock);
384  }
385  else {
386  thismany = atonce;
387  sfs_writedirblock(d + i*atonce, diskblock);
388  }
389  left -= thismany;
390  }
391  assert(left == 0);
392 }

397 /* this exists because qsort() doesn't pass a context pointer through */
398 static struct sfs_direntry *global_sortdirs;
399 
400 /*
401  * Compare function for the permutation vector produced by
402  * sfsdir_sort().
403  */
404 static
405 int
406 dirsortfunc(const void *aa, const void *bb)
407 {
408  const int *a = (const int *)aa;
409  const int *b = (const int *)bb;
410  const struct sfs_direntry *ad = &global_sortdirs[*a];
411  const struct sfs_direntry *bd = &global_sortdirs[*b];
412 
413  /* Sort unallocated entries last */
414  if (ad->sfd_ino == SFS_NOINO && bd->sfd_ino == SFS_NOINO) {
415  return 0;
416  }
417  if (ad->sfd_ino == SFS_NOINO) {
418  return 1;
419  }
420  if (bd->sfd_ino == SFS_NOINO) {
421  return -1;
422  }
423 
424  return strcmp(ad->sfd_name, bd->sfd_name);
425 }

427 /*
428  * Sort the directory contents in D (with ND entries) by producing a
429  * permutation vector into VECTOR, which should be allocated to hold
430  * ND ints.
431  */
432 void
433 sfsdir_sort(struct sfs_direntry *d, unsigned nd, int *vector)
434 {
435  unsigned i;
436 
437  for (i=0; i<nd; i++) {
438  vector[i] = i;
439  }
440 
441  global_sortdirs = d;
442  qsort(vector, nd, sizeof(int), dirsortfunc);
443 }
