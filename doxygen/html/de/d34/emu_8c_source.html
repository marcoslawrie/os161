30 /*
31  * Emulator passthrough filesystem.
32  *
33  * The idea is that this appears as a filesystem in the VFS layer, and
34  * passes VFS operations through a somewhat complicated "hardware"
35  * interface to some simulated "hardware" in System/161 that accesses
36  * the filesystem System/161 is running in.
37  *
38  * This makes it unnecessary to copy the system files to the simulated
39  * disk, although we recommend doing so and trying running without this
40  * device as part of testing your filesystem.
41  */

135 /*
136  * Convert the error codes reported by the "hardware" to errnos.
137  * Or, on cases that indicate a programming error in emu.c, panic.
138  */
139 static
140 uint32_t
141 translate_err(struct emu_softc *sc, uint32_t code)
142 {
143  switch (code) {
144  case EMU_RES_SUCCESS: return 0;
145  case EMU_RES_BADHANDLE:
146  case EMU_RES_BADOP:
147  case EMU_RES_BADSIZE:
148  panic("emu%d: got fatal result code %d\n", sc->e_unit, code);
149  case EMU_RES_BADPATH: return ENOENT;
150  case EMU_RES_EXISTS: return EEXIST;
151  case EMU_RES_ISDIR: return EISDIR;
152  case EMU_RES_MEDIA: return EIO;
153  case EMU_RES_NOHANDLES: return ENFILE;
154  case EMU_RES_NOSPACE: return ENOSPC;
155  case EMU_RES_NOTDIR: return ENOTDIR;
156  case EMU_RES_UNKNOWN: return EIO;
157  case EMU_RES_UNSUPP: return ENOSYS;
158  }
159  kprintf("emu%d: Unknown result code %d\n", sc->e_unit, code);
160  return EAGAIN;
161 }

174 /*
175  * Common file open routine (for both VOP_LOOKUP and VOP_CREATE). Not
176  * for VOP_EACHOPEN. At the hardware level, we need to "open" files in
177  * order to look at them, so by the time VOP_EACHOPEN is called the
178  * files are already open.
179  */
180 static
181 int
182 emu_open(struct emu_softc *sc, uint32_t handle, const char *name,
183  bool create, bool excl, mode_t mode,
184  uint32_t *newhandle, int *newisdir)
185 {
186  uint32_t op;
187  int result;
188 
189  if (strlen(name)+1 > EMU_MAXIO) {
190  return ENAMETOOLONG;
191  }
192 
193  if (create && excl) {
194  op = EMU_OP_EXCLCREATE;
195  }
196  else if (create) {
197  op = EMU_OP_CREATE;
198  }
199  else {
200  op = EMU_OP_OPEN;
201  }
202 
203  /* mode isn't supported (yet?) */
204  (void)mode;
205 
206  lock_acquire(sc->e_lock);
207 
208  strcpy(sc->e_iobuf, name);
209  membar_store_store();
210  emu_wreg(sc, REG_IOLEN, strlen(name));
211  emu_wreg(sc, REG_HANDLE, handle);
212  emu_wreg(sc, REG_OPER, op);
213  result = emu_waitdone(sc);
214 
215  if (result==0) {
216  *newhandle = emu_rreg(sc, REG_HANDLE);
217  *newisdir = emu_rreg(sc, REG_IOLEN)>0;
218  }
219 
220  lock_release(sc->e_lock);
221  return result;
222 }

224 /*
225  * Routine for closing a file we opened at the hardware level.
226  * This is not necessarily called at VOP_LASTCLOSE time; it's called
227  * at VOP_RECLAIM time.
228  */
229 static
230 int
231 emu_close(struct emu_softc *sc, uint32_t handle)
232 {
233  int result;
234  bool mine;
235  int retries = 0;
236 
237  mine = lock_do_i_hold(sc->e_lock);
238  if (!mine) {
239  lock_acquire(sc->e_lock);
240  }
241 
242  while (1) {
243  /* Retry operation up to 10 times */
244 
245  emu_wreg(sc, REG_HANDLE, handle);
246  emu_wreg(sc, REG_OPER, EMU_OP_CLOSE);
247  result = emu_waitdone(sc);
248 
249  if (result==EIO && retries < 10) {
250  kprintf("emu%d: I/O error on close, retrying\n",
251  sc->e_unit);
252  retries++;
253  continue;
254  }
255  break;
256  }
257 
258  if (!mine) {
259  lock_release(sc->e_lock);
260  }
261  return result;
262 }

264 /*
265  * Common code for read and readdir.
266  */
267 static
268 int
269 emu_doread(struct emu_softc *sc, uint32_t handle, uint32_t len,
270  uint32_t op, struct uio *uio)
271 {
272  int result;
273 
274  KASSERT(uio->uio_rw == UIO_READ);
275 
276  if (uio->uio_offset > (off_t)0xffffffff) {
277  /* beyond the largest size the file can have; generate EOF */
278  return 0;
279  }
280 
281  lock_acquire(sc->e_lock);
282 
283  emu_wreg(sc, REG_HANDLE, handle);
284  emu_wreg(sc, REG_IOLEN, len);
285  emu_wreg(sc, REG_OFFSET, uio->uio_offset);
286  emu_wreg(sc, REG_OPER, op);
287  result = emu_waitdone(sc);
288  if (result) {
289  goto out;
290  }
291 
292  membar_load_load();
293  result = uiomove(sc->e_iobuf, emu_rreg(sc, REG_IOLEN), uio);
294 
295  uio->uio_offset = emu_rreg(sc, REG_OFFSET);
296 
297  out:
298  lock_release(sc->e_lock);
299  return result;
300 }

324 /*
325  * Write to a hardware-level file handle.
326  */
327 static
328 int
329 emu_write(struct emu_softc *sc, uint32_t handle, uint32_t len,
330  struct uio *uio)
331 {
332  int result;
333 
334  KASSERT(uio->uio_rw == UIO_WRITE);
335 
336  if (uio->uio_offset > (off_t)0xffffffff) {
337  return EFBIG;
338  }
339 
340  lock_acquire(sc->e_lock);
341 
342  emu_wreg(sc, REG_HANDLE, handle);
343  emu_wreg(sc, REG_IOLEN, len);
344  emu_wreg(sc, REG_OFFSET, uio->uio_offset);
345 
346  result = uiomove(sc->e_iobuf, len, uio);
347  membar_store_store();
348  if (result) {
349  goto out;
350  }
351 
352  emu_wreg(sc, REG_OPER, EMU_OP_WRITE);
353  result = emu_waitdone(sc);
354 
355  out:
356  lock_release(sc->e_lock);
357  return result;
358 }

360 /*
361  * Get the file size associated with a hardware-level file handle.
362  */
363 static
364 int
365 emu_getsize(struct emu_softc *sc, uint32_t handle, off_t *retval)
366 {
367  int result;
368 
369  lock_acquire(sc->e_lock);
370 
371  emu_wreg(sc, REG_HANDLE, handle);
372  emu_wreg(sc, REG_OPER, EMU_OP_GETSIZE);
373  result = emu_waitdone(sc);
374  if (result==0) {
375  *retval = emu_rreg(sc, REG_IOLEN);
376  }
377 
378  lock_release(sc->e_lock);
379  return result;
380 }

461 /*
462  * VOP_RECLAIM
463  *
464  * Reclaim should make an effort to returning errors other than EBUSY.
465  */
466 static
467 int
468 emufs_reclaim(struct vnode *v)
469 {
470  struct emufs_vnode *ev = v->vn_data;
471  struct emufs_fs *ef = v->vn_fs->fs_data;
472  unsigned ix, i, num;
473  int result;
474 
475  /*
476  * Need all of these locks, e_lock to protect the device,
477  * vfs_biglock to protect the fs-related material, and
478  * vn_countlock for the reference count.
479  */
480 
481  vfs_biglock_acquire();
482  lock_acquire(ef->ef_emu->e_lock);
483  spinlock_acquire(&ev->ev_v.vn_countlock);
484 
485  if (ev->ev_v.vn_refcount > 1) {
486  /* consume the reference VOP_DECREF passed us */
487  ev->ev_v.vn_refcount--;
488 
489  spinlock_release(&ev->ev_v.vn_countlock);
490  lock_release(ef->ef_emu->e_lock);
491  vfs_biglock_release();
492  return EBUSY;
493  }
494  KASSERT(ev->ev_v.vn_refcount == 1);
495 
496  /*
497  * Since we hold e_lock and are the last ref, nobody can increment
498  * the refcount, so we can release vn_countlock.
499  */
500  spinlock_release(&ev->ev_v.vn_countlock);
501 
502  /* emu_close retries on I/O error */
503  result = emu_close(ev->ev_emu, ev->ev_handle);
504  if (result) {
505  lock_release(ef->ef_emu->e_lock);
506  vfs_biglock_release();
507  return result;
508  }
509 
510  num = vnodearray_num(ef->ef_vnodes);
511  ix = num;
512  for (i=0; i<num; i++) {
513  struct vnode *vx;
514 
515  vx = vnodearray_get(ef->ef_vnodes, i);
516  if (vx == v) {
517  ix = i;
518  break;
519  }
520  }
521  if (ix == num) {
522  panic("emu%d: reclaim vnode %u not in vnode pool\n",
523  ef->ef_emu->e_unit, ev->ev_handle);
524  }
525 
526  vnodearray_remove(ef->ef_vnodes, ix);
527  vnode_cleanup(&ev->ev_v);
528 
529  lock_release(ef->ef_emu->e_lock);
530  vfs_biglock_release();
531 
532  kfree(ev);
533  return 0;
534 }

536 /*
537  * VOP_READ
538  */
539 static
540 int
541 emufs_read(struct vnode *v, struct uio *uio)
542 {
543  struct emufs_vnode *ev = v->vn_data;
544  uint32_t amt;
545  size_t oldresid;
546  int result;
547 
548  KASSERT(uio->uio_rw==UIO_READ);
549 
550  while (uio->uio_resid > 0) {
551  amt = uio->uio_resid;
552  if (amt > EMU_MAXIO) {
553  amt = EMU_MAXIO;
554  }
555 
556  oldresid = uio->uio_resid;
557 
558  result = emu_read(ev->ev_emu, ev->ev_handle, amt, uio);
559  if (result) {
560  return result;
561  }
562 
563  if (uio->uio_resid == oldresid) {
564  /* nothing read - EOF */
565  break;
566  }
567  }
568 
569  return 0;
570 }
