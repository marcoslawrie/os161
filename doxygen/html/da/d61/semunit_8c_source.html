39 /*
40  * Unit tests for semaphores.
41  *
42  * We test 21 correctness criteria, each stated in a comment at the
43  * top of each test.
44  *
45  * Note that these tests go inside the semaphore abstraction to
46  * validate the internal state.
47  *
48  * All tests (apart from those that crash) attempt to clean up after
49  * running, to avoid leaking memory and leaving extra threads lying
50  * around. Tests with a cleanup phase call ok() before going to it in
51  * case the cleanup crashes -- this should not happen of course but if
52  * it does it should be distinguished from the main part of the test
53  * itself dying.
54  */

71 /*
72  * Wrapper for sem_create when we aren't explicitly tweaking it.
73  */
74 static
75 struct semaphore *
76 makesem(unsigned count)
77 {
78  struct semaphore *sem;
79 
80  sem = sem_create(NAMESTRING, count);
81  if (sem == NULL) {
82  panic("semunit: whoops: sem_create failed\n");
83  }
84  return sem;
85 }

87 /*
88  * A thread that just waits on a semaphore.
89  */
90 static
91 void
92 waiter(void *vsem, unsigned long junk)
93 {
94  struct semaphore *sem = vsem;
95  (void)junk;
96 
97  P(sem);
98 
99  spinlock_acquire(&waiters_lock);
100  KASSERT(waiters_running > 0);
101  waiters_running--;
102  spinlock_release(&waiters_lock);
103 }

105 /*
106  * Set up a waiter.
107  */
108 static
109 void
110 makewaiter(struct semaphore *sem)
111 {
112  int result;
113 
114  spinlock_acquire(&waiters_lock);
115  waiters_running++;
116  spinlock_release(&waiters_lock);
117 
118  result = thread_fork("semunit waiter", NULL, waiter, sem, 0);
119  if (result) {
120  panic("semunit: thread_fork failed\n");
121  }
122  kprintf("Sleeping for waiter to run\n");
123  clocksleep(1);
124 }

126 /*
127  * Spinlocks don't natively provide this, because it only makes sense
128  * under controlled conditions.
129  *
130  * Note that we should really read the holder atomically; but because
131  * we're using this under controlled conditions, it doesn't actually
132  * matter -- nobody is supposed to be able to touch the holder while
133  * we're checking it, or the check wouldn't be reliable; and, provided
134  * clocksleep works, nobody can.
135  */
136 static
137 bool
138 spinlock_not_held(struct spinlock *splk)
139 {
140  return splk->splk_holder == NULL;
141 }

146 /*
147  * 1. After a successful sem_create:
148  * - sem_name compares equal to the passed-in name
149  * - sem_name is not the same pointer as the passed-in name
150  * - sem_wchan is not null
151  * - sem_lock is not held and has no owner
152  * - sem_count is the passed-in count
153  */
154 int
155 semu1(int nargs, char **args)
156 {

159

161

171 
172  ok();
173  /* clean up */
174  sem_destroy(sem);
175  return 0;
176 }

276  * 7. Calling V on a semaphore does not block the caller, regardless
277  * of the semaphore count.

279 int
280 semu7(int nargs, char **args)

284

293  spinlock_init(&lk);

309 /*
310  * 8/9. After calling V on a semaphore with no threads waiting:
311  * - sem_name is unchanged
312  * - sem_wchan is unchanged
313  * - sem_lock is (still) unheld and has no owner
314  * - sem_count is increased by one
315  *
316  * This is true even if we are in an interrupt handler.
317  */
318 static
319 void
320 do_semu89(bool interrupthandler)
321 {
322  struct semaphore *sem;
323  struct wchan *wchan;
324  const char *name;
325 
326  sem = makesem(0);
327 
328  /* check preconditions */
329  name = sem->sem_name;
330  wchan = sem->sem_wchan;
331  KASSERT(!strcmp(name, NAMESTRING));
332  KASSERT(spinlock_not_held(&sem->sem_lock));
333 
334  /*
335  * The right way to this is to set up an actual interrupt,
336  * e.g. an interprocessor interrupt, and hook onto it to run
337  * the V() in the actual interrupt handler. However, that
338  * requires a good bit of infrastructure that we don't
339  * have. Instead we'll fake it by explicitly setting
340  * curthread->t_in_interrupt.
341  */
342  if (interrupthandler) {
343  KASSERT(curthread->t_in_interrupt == false);
344  curthread->t_in_interrupt = true;
345  }
346 
347  V(sem);
348 
349  if (interrupthandler) {
350  KASSERT(curthread->t_in_interrupt == true);
351  curthread->t_in_interrupt = false;
352  }
353 
354  /* check postconditions */
355  KASSERT(name == sem->sem_name);
356  KASSERT(!strcmp(name, NAMESTRING));
357  KASSERT(wchan == sem->sem_wchan);
358  KASSERT(spinlock_not_held(&sem->sem_lock));
359  KASSERT(sem->sem_count == 1);
360 
361  /* clean up */
362  ok();
363  sem_destroy(sem);
364 }
