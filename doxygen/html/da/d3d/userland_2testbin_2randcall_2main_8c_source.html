38 randchar(char *c)
39 {
40 #if RAND_MAX != 0x7fffffff
41 #error "This code assumes RAND_MAX is 0x7fffffff"
42 #endif
43 
44  static long lbits = 0;
45  static long lnum = 0;
46 
47  long bit;
48  int ct = 0;
49 
50  *c = 0;
51 
52  while (ct < CHAR_BIT) {
53  if (lnum==0) {
54  lbits = random();
55  lnum = 31;
56  }
57 
58  bit = lbits & 1;
59  if (bit) {
60  (*c) |= 1;
61  }
62  (*c) <<= 1;
63  ct++;
64  lbits >>= 1;
65  lnum--;
66  }
67 }
