45 void
46 wait_badpid(pid_t pid, const char *desc)
47 {
48  pid_t rv;
49  int x;
50  int err;
51 
52  report_begin(desc);
53  rv = waitpid(pid, &x, 0);
54  err = errno;
55  /* Allow ENOSYS for 0 or negative values of pid only */
56  if (pid <= 0 && rv == -1 && err == ENOSYS) {
57  err = ESRCH;
58  }
59  else if (err == ENOSYS) {
60  report_saw_enosys();
61  }
62  report_check2(rv, err, ESRCH, ECHILD);
63 }

216 static
217 void
218 wait_siblings_child(const char *semname)
219 {
220  pid_t pids[2], mypid, otherpid;
221  int rv, fd, semfd, x;
222  char c;
223 
224  mypid = getpid();
225 
226  /*
227  * Get our own handle for the semaphore, in case naive
228  * file-level synchronization causes concurrent use to
229  * deadlock.
230  */
231  semfd = open(semname, O_RDONLY);
232  if (semfd < 0) {
233  report_warn("child process (pid %d) can't open %s",
234  mypid, semname);
235  }
236  else {
237  if (read(semfd, &c, 1) < 0) {
238  report_warn("in pid %d: %s: read", mypid, semname);
239  }
240  close(semfd);
241  }
242 
243  fd = open(TESTFILE, O_RDONLY);
244  if (fd<0) {
245  report_warn("child process (pid %d) can't open %s",
246  mypid, TESTFILE);
247  return;
248  }
249 
250  /*
251  * In case the semaphore above didn't work, as a backup
252  * busy-wait until the parent writes the pids into the
253  * file. If the semaphore did work, this shouldn't loop.
254  */
255  do {
256  rv = lseek(fd, 0, SEEK_SET);
257  if (rv<0) {
258  report_warn("child process (pid %d) lseek error",
259  mypid);
260  return;
261  }
262  rv = read(fd, pids, sizeof(pids));
263  if (rv<0) {
264  report_warn("child process (pid %d) read error",
265  mypid);
266  return;
267  }
268  } while (rv < (int)sizeof(pids));
269 
270  if (mypid==pids[0]) {
271  otherpid = pids[1];
272  }
273  else if (mypid==pids[1]) {
274  otherpid = pids[0];
275  }
276  else {
277  report_warn("child process (pid %d) got garbage in comm file",
278  mypid);
279  return;
280  }
281  close(fd);
282 
283  rv = waitpid(otherpid, &x, 0);
284  report_beginsub("sibling (pid %d)", mypid);
285  report_survival(rv, errno);
286 }

289 void
290 wait_siblings(void)
291 {
292  pid_t pids[2];
293  int rv, fd, semfd, x;
294  int bad = 0;
295  char semname[32];
296 
297  /* This test may also blow up if FS synchronization is substandard */
298 
299  report_begin("siblings wait for each other");
300  report_hassubs();
301 
302  snprintf(semname, sizeof(semname), "sem:badcall.%d", (int)getpid());
303  semfd = open(semname, O_WRONLY|O_CREAT|O_TRUNC, 0664);
304  if (semfd < 0) {
305  report_warn("can't make semaphore");
306  report_aborted();
307  return;
308  }
309 
310  fd = open_testfile(NULL);
311  if (fd<0) {
312  report_aborted();
313  close(semfd);
314  remove(semname);
315  return;
316  }
317 
318  pids[0] = fork();
319  if (pids[0]<0) {
320  report_warn("can't fork");
321  report_aborted();
322  close(fd);
323  close(semfd);
324  remove(semname);
325  return;
326  }
327  if (pids[0]==0) {
328  close(fd);
329  close(semfd);
330  wait_siblings_child(semname);
331  _exit(0);
332  }
333 
334  pids[1] = fork();
335  if (pids[1]<0) {
336  report_warn("can't fork");
337  report_aborted();
338  /* abandon the other child process :( */
339  close(fd);
340  close(semfd);
341  remove(semname);
342  return;
343  }
344  if (pids[1]==0) {
345  close(fd);
346  close(semfd);
347  wait_siblings_child(semname);
348  _exit(0);
349  }
350 
351  rv = write(fd, pids, sizeof(pids));
352  if (rv < 0) {
353  report_warn("write error on %s", TESTFILE);
354  report_aborted();
355  /* abandon child procs :( */
356  close(fd);
357  close(semfd);
358  remove(semname);
359  return;
360  }
361  if (rv != (int)sizeof(pids)) {
362  report_warnx("write error on %s: short count", TESTFILE);
363  report_aborted();
364  /* abandon child procs :( */
365  close(fd);
366  close(semfd);
367  remove(semname);
368  return;
369  }
370 
371  /* gate the child procs */
372  rv = write(semfd, " ", 2);
373  if (rv < 0) {
374  report_warn("%s: write", semname);
375  bad = 1;
376  }
377 
378  report_beginsub("overall");
379  rv = waitpid(pids[0], &x, 0);
380  if (rv<0) {
381  report_warn("error waiting for child 0 (pid %d)", pids[0]);
382  bad = 1;
383  }
384  rv = waitpid(pids[1], &x, 0);
385  if (rv<0) {
386  report_warn("error waiting for child 1 (pid %d)", pids[1]);
387  bad = 1;
388  }
389  if (bad) {
390  /* XXX: aborted, or failure, or what? */
391  report_aborted();
392  }
393  else {
394  report_passed();
395  }
396  close(fd);
397  close(semfd);
398  remove(semname);
399  remove(TESTFILE);
400 }
