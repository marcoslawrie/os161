41 /* Print a file that's already been opened. */
42 static
43 void
44 docat(const char *name, int fd)
45 {
46  char buf[1024];
47  int len, wr, wrtot;
48 
49  /*
50  * As long as we get more than zero bytes, we haven't hit EOF.
51  * Zero means EOF. Less than zero means an error occurred.
52  * We may read less than we asked for, though, in various cases
53  * for various reasons.
54  */
55  while ((len = read(fd, buf, sizeof(buf)))>0) {
56  /*
57  * Likewise, we may actually write less than we attempted
58  * to. So loop until we're done.
59  */
60  wrtot = 0;
61  while (wrtot < len) {
62  wr = write(STDOUT_FILENO, buf+wrtot, len-wrtot);
63  if (wr<0) {
64  err(1, "stdout");
65  }
66  wrtot += wr;
67  }
68  }
69  /*
70  * If we got a read error, print it and exit.
71  */
72  if (len<0) {
73  err(1, "%s", name);
74  }
75 }

77 /* Print a file by name. */
78 static
79 void
80 cat(const char *file)
81 {
82  int fd;
83 
84  /*
85  * "-" means print stdin.
86  */
87  if (!strcmp(file, "-")) {
88  docat("stdin", STDIN_FILENO);
89  return;
90  }
91 
92  /*
93  * Open the file, print it, and close it.
94  * Bail out if we can't open it.
95  */
96  fd = open(file, O_RDONLY);
97  if (fd<0) {
98  err(1, "%s", file);
99  }
100  docat(file, fd);
101  close(fd);
102 }

106 main(int argc, char *argv[])
107 {
108  if (argc==1) {
109  /* No args - just do stdin */

112  else {
113  /* Print all the files specified on the command line. */
114  int i;
115  for (i=1; i<argc; i++) {
116  cat(argv[i]);

118  }
