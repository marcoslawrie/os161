45 int
46 open_testfile(const char *string)
47 {
48  int fd, rv;
49  size_t len;
50 
51  fd = open(TESTFILE, O_RDWR|O_CREAT|O_TRUNC, 0664);
52  if (fd<0) {
53  report_warn("creating %s: failed", TESTFILE);
54  return -1;
55  }
56 
57  if (string) {
58  len = strlen(string);
59  rv = write(fd, string, len);
60  if (rv<0) {
61  report_warn("write to %s failed", TESTFILE);
62  close(fd);
63  remove(TESTFILE);
64  return -1;
65  }
66  if ((unsigned)rv != len) {
67  report_warn("write to %s got short count", TESTFILE);
68  close(fd);
69  remove(TESTFILE);
70  return -1;
71  }
72  rv = lseek(fd, 0, SEEK_SET);
73  if (rv<0) {
74  report_warn("rewind of %s failed", TESTFILE);
75  close(fd);
76  remove(TESTFILE);
77  return -1;
78  }
79  }
80  return fd;
81 }

195 static
196 void
197 menu(void)
198 {
199  int i;
200  for (i=0; ops[i].name; i++) {
201  printf("[%c] %-24s", ops[i].ch, ops[i].name);
202  if (i%2==1) {
203  printf("\n");
204  }
205  }
206  if (i%2==1) {
207  printf("\n");
208  }
209  printf("[1] %-24s", "asst1");
210  printf("[2] %-24s\n", "asst2");
211  printf("[3] %-24s", "asst3");

213  printf("[*] %-24s", "all");

217 static
218 void
219 runit(int op)
220 {
221  int i, k;
222 
223  if (op=='!') {
224  exit(0);
225  }
226 
227  if (op=='?') {
228  menu();
229  return;
230  }
231 
232  if (op=='*') {
233  for (i=0; ops[i].name; i++) {
234  printf("[%s]\n", ops[i].name);
235  ops[i].f();
236  }
237  return;
238  }
239 
240  if (op>='1' && op <= '4') {
241  k = op-'0';
242  for (i=0; ops[i].name; i++) {
243  if (ops[i].asst <= k) {
244  printf("[%s]\n", ops[i].name);
245  ops[i].f();
246  }
247  }
248  return;
249  }
250 
251  if (op < LOWEST || op > HIGHEST) {
252  printf("Invalid request %c\n", op);
253  return;
254  }
255 
256  ops[op-'a'].f();
257 }

259 int
260 main(int argc, char **argv)
261 {
262  int op, i, j;
263 
264  printf("[%c-%c, 1-4, *, ?=menu, !=quit]\n", LOWEST, HIGHEST);
265 
266  if (argc > 1) {
267  for (i=1; i<argc; i++) {
268  for (j=0; argv[i][j]; j++) {
269  printf("Choose: %c\n",
270  argv[i][j]);
271  runit(argv[i][j]);
272  }
273  }
274  }
275  else {
276  menu();
277  while (1) {
278  printf("Choose: ");
279  op = getchar();
280  if (op==EOF) {
281  break;
282  }
283  printf("%c\n", op);
284  runit(op);
285  }
286  }
287 
288  return 0;
