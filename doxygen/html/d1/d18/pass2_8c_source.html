50 /*
51  * Process a directory. INO is the inode number; PARENTINO is the
52  * parent's inode number; PATHSOFAR is the path to this directory.
53  *
54  * Recursively checks its subdirs.
55  *
56  * In the FUTURE we might want to improve the handling of crosslinked
57  * directories so it picks the parent that the .. entry points to,
58  * instead of the first entry we recursively find. Beware of course
59  * that the .. entry might not point to anywhere valid at all...
60  */
61 static
62 int
63 pass2_dir(uint32_t ino, uint32_t parentino, const char *pathsofar)
64 {
65  struct sfs_dinode sfi;
66  struct sfs_direntry *direntries;
67  int *sortvector;
68  uint32_t dirsize, ndirentries, maxdirentries, subdircount, i;
69  int ichanged=0, dchanged=0, dotseen=0, dotdotseen=0;
70 
71  if (inode_visitdir(ino)) {
72  /* crosslinked dir; tell parent to remove the entry */
73  return 1;
74  }
75 
76  /* Load the inode. */
77  sfs_readinode(ino, &sfi);
78 
79  /*
80  * Load the directory. If there is any leftover room in the
81  * last block, allocate space for it in case we want to insert
82  * entries.
83  */
84 
85  ndirentries = sfi.sfi_size/sizeof(struct sfs_direntry);
86  maxdirentries = SFS_ROUNDUP(ndirentries,
87  SFS_BLOCKSIZE/sizeof(struct sfs_direntry));
88  dirsize = maxdirentries * sizeof(struct sfs_direntry);
89  direntries = domalloc(dirsize);
90 
91  sortvector = domalloc(ndirentries * sizeof(int));
92 
93  sfs_readdir(&sfi, direntries, ndirentries);
94  for (i=ndirentries; i<maxdirentries; i++) {
95  direntries[i].sfd_ino = SFS_NOINO;
96  bzero(direntries[i].sfd_name, sizeof(direntries[i].sfd_name));
97  }
98 
99  /*
100  * Sort by name and check for duplicate names.
101  */
102 
103  sfsdir_sort(direntries, ndirentries, sortvector);
104 
105  /* don't use ndirentries-1 here, in case ndirentries == 0 */
106  for (i=0; i+1<ndirentries; i++) {
107  struct sfs_direntry *d1 = &direntries[sortvector[i]];
108  struct sfs_direntry *d2 = &direntries[sortvector[i+1]];
109  assert(d1 != d2);
110 
111  if (d1->sfd_ino == SFS_NOINO || d2->sfd_ino == SFS_NOINO) {
112  /* sfsdir_sort puts these last */
113  continue;
114  }
115 
116  if (!strcmp(d1->sfd_name, d2->sfd_name)) {
117  if (d1->sfd_ino == d2->sfd_ino) {
118  setbadness(EXIT_RECOV);
119  warnx("Directory %s: Duplicate entries for "
120  "%s (merged)",
121  pathsofar, d1->sfd_name);
122  d1->sfd_ino = SFS_NOINO;
123  d1->sfd_name[0] = 0;
124  }
125  else {
126  /* XXX: what if FSCK.n.m already exists? */
127  snprintf(d1->sfd_name, sizeof(d1->sfd_name),
128  "FSCK.%lu.%lu",
129  (unsigned long) d1->sfd_ino,
130  (unsigned long) uniqueid());
131  setbadness(EXIT_RECOV);
132  warnx("Directory %s: Duplicate names %s "
133  "(one renamed: %s)",
134  pathsofar, d2->sfd_name, d1->sfd_name);
135  }
136  dchanged = 1;
137  }
138  }
139 
140  /*
141  * Look for the . and .. entries.
142  */
143 
144  for (i=0; i<ndirentries; i++) {
145  if (!strcmp(direntries[i].sfd_name, ".")) {
146  if (direntries[i].sfd_ino != ino) {
147  setbadness(EXIT_RECOV);
148  warnx("Directory %s: Incorrect `.' entry "
149  "(fixed)", pathsofar);
150  direntries[i].sfd_ino = ino;
151  dchanged = 1;
152  }
153  /* duplicates are checked above -> only one . here */
154  assert(dotseen==0);
155  dotseen = 1;
156  }
157  else if (!strcmp(direntries[i].sfd_name, "..")) {
158  if (direntries[i].sfd_ino != parentino) {
159  setbadness(EXIT_RECOV);
160  warnx("Directory %s: Incorrect `..' entry "
161  "(fixed)", pathsofar);
162  direntries[i].sfd_ino = parentino;
163  dchanged = 1;
164  }
165  /* duplicates are checked above -> only one .. here */
166  assert(dotdotseen==0);
167  dotdotseen = 1;
168  }
169  }
170 
171  /*
172  * If no . entry, try to insert one.
173  */
174 
175  if (!dotseen) {
176  if (sfsdir_tryadd(direntries, ndirentries, ".", ino)==0) {
177  setbadness(EXIT_RECOV);
178  warnx("Directory %s: No `.' entry (added)",
179  pathsofar);
180  dchanged = 1;
181  }
182  else if (sfsdir_tryadd(direntries, maxdirentries, ".",
183  ino)==0) {
184  setbadness(EXIT_RECOV);
185  warnx("Directory %s: No `.' entry (added)",
186  pathsofar);
187  ndirentries++;
188  dchanged = 1;
189  sfi.sfi_size += sizeof(struct sfs_direntry);
190  ichanged = 1;
191  }
192  else {
193  setbadness(EXIT_UNRECOV);
194  warnx("Directory %s: No `.' entry (NOT FIXED)",
195  pathsofar);
196  }
197  }
198 
199  /*
200  * If no .. entry, try to insert one.
201  */
202 
203  if (!dotdotseen) {
204  if (sfsdir_tryadd(direntries, ndirentries, "..",
205  parentino)==0) {
206  setbadness(EXIT_RECOV);
207  warnx("Directory %s: No `..' entry (added)",
208  pathsofar);
209  dchanged = 1;
210  }
211  else if (sfsdir_tryadd(direntries, maxdirentries, "..",
212  parentino)==0) {
213  setbadness(EXIT_RECOV);
214  warnx("Directory %s: No `..' entry (added)",
215  pathsofar);
216  ndirentries++;
217  dchanged = 1;
218  sfi.sfi_size += sizeof(struct sfs_direntry);
219  ichanged = 1;
220  }
221  else {
222  setbadness(EXIT_UNRECOV);
223  warnx("Directory %s: No `..' entry (NOT FIXED)",
224  pathsofar);
225  }
226  }
227 
228  /*
229  * Now load each inode in the directory.
230  *
231  * For regular files, count the number of links we see; for
232  * directories, recurse. Count the number of subdirs seen
233  * so we can correct our own link count if necessary.
234  */
235 
236  subdircount=0;
237  for (i=0; i<ndirentries; i++) {
238  if (direntries[i].sfd_ino == SFS_NOINO) {
239  /* nothing */
240  }
241  else if (!strcmp(direntries[i].sfd_name, ".")) {
242  /* nothing */
243  }
244  else if (!strcmp(direntries[i].sfd_name, "..")) {
245  /* nothing */
246  }
247  else {
248  char path[strlen(pathsofar)+SFS_NAMELEN+1];
249  struct sfs_dinode subsfi;
250 
251  sfs_readinode(direntries[i].sfd_ino, &subsfi);
252  snprintf(path, sizeof(path), "%s/%s",
253  pathsofar, direntries[i].sfd_name);
254 
255  switch (subsfi.sfi_type) {
256  case SFS_TYPE_FILE:
257  inode_addlink(direntries[i].sfd_ino);
258  break;
259  case SFS_TYPE_DIR:
260  if (pass2_dir(direntries[i].sfd_ino,
261  ino,
262  path)) {
263  setbadness(EXIT_RECOV);
264  warnx("Directory %s: Crosslink to "
265  "other directory (removed)",
266  path);
267  direntries[i].sfd_ino = SFS_NOINO;
268  direntries[i].sfd_name[0] = 0;
269  dchanged = 1;
270  }
271  else {
272  subdircount++;
273  }
274  break;
275  default:
276  setbadness(EXIT_RECOV);
277  warnx("Object %s: Invalid inode type %u "
278  "(removed)", path, subsfi.sfi_type);
279  direntries[i].sfd_ino = SFS_NOINO;
280  direntries[i].sfd_name[0] = 0;
281  dchanged = 1;
282  break;
283  }
284  }
285  }
286 
287  /*
288  * Fix up the link count if needed.
289  */
290 
291  if (sfi.sfi_linkcount != subdircount+2) {
292  setbadness(EXIT_RECOV);
293  warnx("Directory %s: Link count %lu should be %lu (fixed)",
294  pathsofar, (unsigned long) sfi.sfi_linkcount,
295  (unsigned long) subdircount+2);
296  sfi.sfi_linkcount = subdircount+2;
297  ichanged = 1;
298  }
299 
300  /*
301  * Write back anything that changed, clean up, and return.
302  */
303 
304  if (dchanged) {
305  sfs_writedir(&sfi, direntries, ndirentries);
306  }
307 
308  if (ichanged) {
309  sfs_writeinode(ino, &sfi);
310  }
311 
312  free(direntries);
313  free(sortvector);
314 
315  return 0;
316 }
