30 /*
31  * User-level malloc and free implementation.
32  *
33  * This is a basic first-fit allocator. It's intended to be simple and
34  * easy to follow. It performs abysmally if the heap becomes larger than
35  * physical memory. To get (much) better out-of-core performance, port
36  * the kernel's malloc. :-)
37  */

55 /*
56  * malloc block header.
57  *
58  * mh_prevblock is the downwards offset to the previous header, 0 if this
59  * is the bottom of the heap.
60  *
61  * mh_nextblock is the upwards offset to the next header.
62  *
63  * mh_pad is unused.
64  * mh_inuse is 1 if the block is in use, 0 if it is free.
65  * mh_magic* should always be a fixed value.
66  *
67  * MBLOCKSIZE should equal sizeof(struct mheader) and be a power of 2.
68  * MBLOCKSHIFT is the log base 2 of MBLOCKSIZE.
69  * MMAGIC is the value for mh_magic*.
70  */
71 struct mheader {
72 
73 #if defined(MALLOC32)
74 #define MBLOCKSIZE 8
75 #define MBLOCKSHIFT 3
76 #define MMAGIC 2
77  /*
78  * 32-bit platform. size_t is 32 bits (4 bytes).
79  * Block size is 8 bytes.
80  */
81  unsigned mh_prevblock:29;
82  unsigned mh_pad:1;
83  unsigned mh_magic1:2;
84 
85  unsigned mh_nextblock:29;
86  unsigned mh_inuse:1;
87  unsigned mh_magic2:2;
88 
89 #elif defined(MALLOC64)
90 #define MBLOCKSIZE 16
91 #define MBLOCKSHIFT 4
92 #define MMAGIC 6
93  /*
94  * 64-bit platform. size_t is 64 bits (8 bytes)
95  * Block size is 16 bytes.
96  */
97  unsigned mh_prevblock:60;
98  unsigned mh_pad:1;
99  unsigned mh_magic1:3;
100 
101  unsigned mh_nextblock:60;
102  unsigned mh_inuse:1;
103  unsigned mh_magic2:3;
104 
105 #else
106 #error "please fix me"
107 #endif
108 };

110 /*
111  * Operator macros on struct mheader.
112  *
113  * M_NEXT/PREVOFF: return offset to next/previous header
114  * M_NEXT/PREV: return next/previous header
115  *
116  * M_DATA: return data pointer of a header
117  * M_SIZE: return data size of a header
118  *
119  * M_OK: true if the magic values are correct
120  *
121  * M_MKFIELD: prepare a value for mh_next/prevblock.
122  * (value should include the header size)
123  */
124 
125 #define M_NEXTOFF(mh) ((size_t)(((size_t)((mh)->mh_nextblock))<<MBLOCKSHIFT))
126 #define M_PREVOFF(mh) ((size_t)(((size_t)((mh)->mh_prevblock))<<MBLOCKSHIFT))
127 #define M_NEXT(mh) ((struct mheader *)(((char*)(mh))+M_NEXTOFF(mh)))
128 #define M_PREV(mh) ((struct mheader *)(((char*)(mh))-M_PREVOFF(mh)))
129 
130 #define M_DATA(mh) ((void *)((mh)+1))
131 #define M_SIZE(mh) (M_NEXTOFF(mh)-MBLOCKSIZE)
132 
133 #define M_OK(mh) ((mh)->mh_magic1==MMAGIC && (mh)->mh_magic2==MMAGIC)
134 
135 #define M_MKFIELD(off) ((off)>>MBLOCKSHIFT)

157 /*
158  * Setup function.
159  */
160 static
161 void
162 __malloc_init(void)
163 {
164  void *x;
165 
166  /*
167  * Check various assumed properties of the sizes.
168  */
169  if (sizeof(struct mheader) != MBLOCKSIZE) {
170  errx(1, "malloc: Internal error - MBLOCKSIZE wrong");
171  }
172  if ((MBLOCKSIZE & (MBLOCKSIZE-1))!=0) {
173  errx(1, "malloc: Internal error - MBLOCKSIZE not power of 2");
174  }
175  if (1<<MBLOCKSHIFT != MBLOCKSIZE) {
176  errx(1, "malloc: Internal error - MBLOCKSHIFT wrong");
177  }
178 
179  /* init should only be called once. */
180  if (__heapbase!=0 || __heaptop!=0) {
181  errx(1, "malloc: Internal error - bad init call");
182  }
183 
184  /* Get the page size, if needed. */
185 #ifdef _SC_PAGESIZE
186  __malloc_pagesize = sysconf(_SC_PAGESIZE);
187 #endif
188 
189  /* Use sbrk to find the base of the heap. */
190  x = sbrk(0);
191  if (x==(void *)-1) {
192  err(1, "malloc: initial sbrk failed");
193  }
194  if (x==(void *) 0) {
195  errx(1, "malloc: Internal error - heap began at 0");
196  }
197  __heapbase = __heaptop = (uintptr_t)x;
198 
199  /*
200  * Make sure the heap base is aligned the way we want it.
201  * (On OS/161, it will begin on a page boundary. But on
202  * an arbitrary Unix, it may not be, as traditionally it
203  * begins at _end.)
204  */
205 
206  if (__heapbase % MBLOCKSIZE != 0) {
207  size_t adjust = MBLOCKSIZE - (__heapbase % MBLOCKSIZE);
208  x = sbrk(adjust);
209  if (x==(void *)-1) {
210  err(1, "malloc: sbrk failed aligning heap base");
211  }
212  if ((uintptr_t)x != __heapbase) {
213  err(1, "malloc: heap base moved during init");
214  }
215 #ifdef MALLOCDEBUG
216  warnx("malloc: adjusted heap base upwards by %lu bytes",
217  (unsigned long) adjust);
218 #endif
219  __heapbase += adjust;
220  __heaptop = __heapbase;
221  }
222 }

276 /*
277  * Get more memory (at the top of the heap) using sbrk, and
278  * return a pointer to it.
279  */
280 static
281 void *
282 __malloc_sbrk(size_t size)
283 {
284  void *x;
285 
286  x = sbrk(size);
287  if (x == (void *)-1) {
288  return NULL;
289  }
290 
291  if ((uintptr_t)x != __heaptop) {
292  errx(1, "malloc: Internal error - "
293  "heap top moved itself from 0x%lx to 0x%lx",
294  (unsigned long) __heaptop,
295  (unsigned long) (uintptr_t) x);
296  }
297  __heaptop += size;
298  return x;
299 }

301 /*
302  * Make a new (free) block from the block passed in, leaving size
303  * bytes for data in the current block. size must be a multiple of
304  * MBLOCKSIZE.
305  *
306  * Only split if the excess space is at least twice the blocksize -
307  * one blocksize to hold a header and one for data.
308  */
309 static
310 void
311 __malloc_split(struct mheader *mh, size_t size)
312 {
313  struct mheader *mhnext, *mhnew;
314  size_t oldsize;
315 
316  if (size % MBLOCKSIZE != 0) {
317  errx(1, "malloc: Internal error (size %lu passed to split)",
318  (unsigned long) size);
319  }
320 
321  if (M_SIZE(mh) - size < 2*MBLOCKSIZE) {
322  /* no room */
323  return;
324  }
325 
326  mhnext = M_NEXT(mh);
327 
328  oldsize = M_SIZE(mh);
329  mh->mh_nextblock = M_MKFIELD(size + MBLOCKSIZE);
330 
331  mhnew = M_NEXT(mh);
332  if (mhnew==mhnext) {
333  errx(1, "malloc: Internal error (split screwed up?)");
334  }
335 
336  mhnew->mh_prevblock = M_MKFIELD(size + MBLOCKSIZE);
337  mhnew->mh_pad = 0;
338  mhnew->mh_magic1 = MMAGIC;
339  mhnew->mh_nextblock = M_MKFIELD(oldsize - size);
340  mhnew->mh_inuse = 0;
341  mhnew->mh_magic2 = MMAGIC;
342 
343  if (mhnext != (struct mheader *) __heaptop) {
344  mhnext->mh_prevblock = mhnew->mh_nextblock;
345  }
346 }

348 /*
349  * malloc itself.
350  */
351 void *
352 malloc(size_t size)
353 {
354  struct mheader *mh;
355  uintptr_t i;
356  size_t rightprevblock;
357  size_t morespace;
358  void *p;
359 
360  if (__heapbase==0) {
361  __malloc_init();
362  }
363  if (__heapbase==0 || __heaptop==0 || __heapbase > __heaptop) {
364  warnx("malloc: Internal error - local data corrupt");
365  errx(1, "malloc: heapbase 0x%lx; heaptop 0x%lx",
366  (unsigned long) __heapbase, (unsigned long) __heaptop);
367  }
368 
369 #ifdef MALLOCDEBUG
370  warnx("malloc: about to allocate %lu (0x%lx) bytes",
371  (unsigned long) size, (unsigned long) size);
372  __malloc_dump();
373 #endif
374 
375  /* Round size up to an integral number of blocks. */
376  size = ((size + MBLOCKSIZE - 1) & ~(size_t)(MBLOCKSIZE-1));
377 
378  /*
379  * First-fit search algorithm for available blocks.
380  * Check to make sure the next/previous sizes all agree.
381  */
382  rightprevblock = 0;
383  mh = NULL;
384  for (i=__heapbase; i<__heaptop; i += M_NEXTOFF(mh)) {
385  mh = (struct mheader *) i;
386  if (!M_OK(mh)) {
387  errx(1, "malloc: Heap corrupt; header at 0x%lx"
388  " has bad magic bits",
389  (unsigned long) i);
390  }
391  if (mh->mh_prevblock != rightprevblock) {
392  errx(1, "malloc: Heap corrupt; header at 0x%lx"
393  " has bad previous-block size %lu "
394  "(should be %lu)",
395  (unsigned long) i,
396  (unsigned long) mh->mh_prevblock << MBLOCKSHIFT,
397  (unsigned long) rightprevblock << MBLOCKSHIFT);
398  }
399  rightprevblock = mh->mh_nextblock;
400 
401  /* Can't allocate a block that's in use. */
402  if (mh->mh_inuse) {
403  continue;
404  }
405 
406  /* Can't allocate a block that isn't big enough. */
407  if (M_SIZE(mh) < size) {
408  continue;
409  }
410 
411  /* Try splitting block. */
412  __malloc_split(mh, size);
413 
414  /*
415  * Now, allocate.
416  */
417  mh->mh_inuse = 1;
418 
419 #ifdef MALLOCDEBUG
420  warnx("malloc: allocating at %p", M_DATA(mh));
421  __malloc_dump();
422 #endif
423  return M_DATA(mh);
424  }
425  if (i!=__heaptop) {
426  errx(1, "malloc: Heap corrupt; ran off end");
427  }
428 
429  /*
430  * Didn't find anything. Expand the heap.
431  *
432  * If the heap is nonempty and the top block (the one mh is
433  * left pointing to after the above loop) is free, we can
434  * expand it. Otherwise we need a new block.
435  */
436  if (mh != NULL && !mh->mh_inuse) {
437  assert(size > M_SIZE(mh));
438  morespace = size - M_SIZE(mh);
439  }
440  else {
441  morespace = MBLOCKSIZE + size;
442  }
443 
444  /* Round the amount of space we ask for up to a whole page. */
445  morespace = PAGE_SIZE * ((morespace + PAGE_SIZE - 1) / PAGE_SIZE);
446 
447  p = __malloc_sbrk(morespace);
448  if (p == NULL) {
449  return NULL;
450  }
451 
452  if (mh != NULL && !mh->mh_inuse) {
453  /* update old header */
454  mh->mh_nextblock = M_MKFIELD(M_NEXTOFF(mh) + morespace);
455  mh->mh_inuse = 1;
456  }
457  else {
458  /* fill out new header */
459  mh = p;
460  mh->mh_prevblock = rightprevblock;
461  mh->mh_magic1 = MMAGIC;
462  mh->mh_magic2 = MMAGIC;
463  mh->mh_pad = 0;
464  mh->mh_inuse = 1;
465  mh->mh_nextblock = M_MKFIELD(morespace);
466  }
467 
468  /*
469  * Either way, try splitting the block we got as because of
470  * the page rounding it might be quite a bit bigger than we
471  * needed.
472  */
473  __malloc_split(mh, size);
474 
475 #ifdef MALLOCDEBUG
476  warnx("malloc: allocating at %p", M_DATA(mh));
477  __malloc_dump();
478 #endif
479  return M_DATA(mh);
480 }

499 /*
500  * Attempt to merge two adjacent blocks (mh below mhnext).
501  */
502 static
503 void
504 __malloc_trymerge(struct mheader *mh, struct mheader *mhnext)
505 {
506  struct mheader *mhnextnext;
507 
508  if (mh->mh_nextblock != mhnext->mh_prevblock) {
509  errx(1, "free: Heap corrupt (%p and %p inconsistent)",
510  mh, mhnext);
511  }
512  if (mh->mh_inuse || mhnext->mh_inuse) {
513  /* can't merge */
514  return;
515  }
516 
517  mhnextnext = M_NEXT(mhnext);
518 
519  mh->mh_nextblock = M_MKFIELD(MBLOCKSIZE + M_SIZE(mh) +
520  MBLOCKSIZE + M_SIZE(mhnext));
521 
522  if (mhnextnext != (struct mheader *)__heaptop) {
523  mhnextnext->mh_prevblock = mh->mh_nextblock;
524  }
525 
526  /* Deadbeef out the memory used by the now-obsolete header */
527  __malloc_deadbeef(mhnext, sizeof(struct mheader));
528 }

530 /*
531  * The actual free() implementation.
532  */
533 void
534 free(void *x)
535 {
536  struct mheader *mh, *mhnext, *mhprev;
537 
538  if (x==NULL) {
539  /* safest practice */
540  return;
541  }
542 
543  /* Consistency check. */
544  if (__heapbase==0 || __heaptop==0 || __heapbase > __heaptop) {
545  warnx("free: Internal error - local data corrupt");
546  errx(1, "free: heapbase 0x%lx; heaptop 0x%lx",
547  (unsigned long) __heapbase, (unsigned long) __heaptop);
548  }
549 
550  /* Don't allow freeing pointers that aren't on the heap. */
551  if ((uintptr_t)x < __heapbase || (uintptr_t)x >= __heaptop) {
552  errx(1, "free: Invalid pointer %p freed (out of range)", x);
553  }
554 
555 #ifdef MALLOCDEBUG
556  warnx("free: about to free %p", x);
557  __malloc_dump();
558 #endif
559 
560  mh = ((struct mheader *)x)-1;
561  if (!M_OK(mh)) {
562  errx(1, "free: Invalid pointer %p freed (corrupt header)", x);
563  }
564 
565  if (!mh->mh_inuse) {
566  errx(1, "free: Invalid pointer %p freed (already free)", x);
567  }
568 
569  /* mark it free */
570  mh->mh_inuse = 0;
571 
572  /* wipe it */
573  __malloc_deadbeef(M_DATA(mh), M_SIZE(mh));
574 
575  /* Try merging with the block above (but not if we're at the top) */
576  mhnext = M_NEXT(mh);
577  if (mhnext != (struct mheader *)__heaptop) {
578  __malloc_trymerge(mh, mhnext);
579  }
580 
581  /* Try merging with the block below (but not if we're at the bottom) */
582  if (mh != (struct mheader *)__heapbase) {
583  mhprev = M_PREV(mh);
584  __malloc_trymerge(mhprev, mh);
585  }
586 
587 #ifdef MALLOCDEBUG
588  warnx("free: freed %p", x);
589  __malloc_dump();
590 #endif
591 }
