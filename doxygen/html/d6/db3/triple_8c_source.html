41 static
42 pid_t
43 spawnv(const char *prog, char **argv)
44 {
45  pid_t pid = fork();
46  switch (pid) {
47  case -1:
48  err(1, "fork");
49  case 0:
50  /* child */
51  execv(prog, argv);
52  err(1, "%s: execv", prog);
53  default:
54  /* parent */
55  break;
56  }
57  return pid;
58 }

60 static
61 int
62 dowait(int index, int pid)
63 {
64  int status;
65 
66  if (waitpid(pid, &status, 0)<0) {
67  warn("waitpid for copy #%d (pid %d)", index, pid);
68  return 1;
69  }
70  else if (WIFSIGNALED(status)) {
71  warnx("copy #%d (pid %d): signal %d", index, pid,
72  WTERMSIG(status));
73  return 1;
74  }
75  else if (WEXITSTATUS(status) != 0) {
76  warnx("copy #%d (pid %d): exit %d", index, pid,
77  WEXITSTATUS(status));
78  return 1;
79  }
80  return 0;
81 }

83 void
84 triple(const char *prog)
85 {
86  pid_t pids[3];
87  int i, failures = 0;
88  char *args[2];
89 
90  /* set up the argv */
91  args[0]=(char *)prog;
92  args[1]=NULL;
93 
94  warnx("Starting: running three copies of %s...", prog);
95 
96  for (i=0; i<3; i++) {
97  pids[i]=spawnv(args[0], args);
98  }
99 
100  for (i=0; i<3; i++) {
101  failures += dowait(i, pids[i]);
102  }
103 
104  if (failures > 0) {
105  warnx("%d failures", failures);
106  }
107  else {
108  warnx("Congratulations! You passed.");
109  }
110 }
