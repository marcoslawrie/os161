159  /* Silence warning with gcc 4.8 -Og (but not -O2) */

162  kprintf("Beginning large array test...\n");
163  a = array_create();
164  KASSERT(a != NULL);
165 
166  /* 1. Fill it one at a time. */
167  p = (void *)0xc0ffee;
168  for (i=0; i<BIGTESTSIZE; i++) {
169  result = array_add(a, p, &x);
170  KASSERT(result == 0);
171  KASSERT(x == i);
172  }
173  KASSERT(array_num(a) == BIGTESTSIZE);
174 
175  /* 2. Check the contents */
176  for (i=0; i<BIGTESTSIZE; i++) {
177  KASSERT(array_get(a, i) == p);
178  }
179 
180  /* 3. Clear it */
181  result = array_setsize(a, 0);
182  KASSERT(result == 0);
183 
184  /* 4. Set the size and initialize with array_set */
185  result = array_setsize(a, BIGTESTSIZE);
186  KASSERT(result == 0);
187  for (i=0; i<BIGTESTSIZE; i++) {
188  array_set(a, i, NTH(i));
189  }
190 
191  /* 5. Check the contents again */
192  for (i=0; i<BIGTESTSIZE; i++) {
193  KASSERT(array_get(a, i) == NTH(i));
194  }
195 
196  /* 6. Zot an entry and check the contents */
197  array_remove(a, 1);
198  KASSERT(array_get(a, 0) == NTH(0));
199  KASSERT(array_num(a) == BIGTESTSIZE-1);
200  for (i=1; i<BIGTESTSIZE-1; i++) {
201  KASSERT(array_get(a, i) == NTH(i+1));
202  }
203 
204  /* 7. Double the size and check the preexisting contents */
205  result = array_setsize(a, BIGTESTSIZE*2);
206  KASSERT(result == 0);
207  KASSERT(array_get(a, 0) == NTH(0));
208  for (i=1; i<BIGTESTSIZE-1; i++) {
209  KASSERT(array_get(a, i) == NTH(i+1));
210  }
211 
212  /* done */
213  result = array_setsize(a, 0);
214  KASSERT(result == 0);
215  array_destroy(a);
216 
217  kprintf("Done.\n");
218 
219  return 0;
220 }
