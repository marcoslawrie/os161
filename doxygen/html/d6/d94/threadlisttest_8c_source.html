54 /*
55  * Create a dummy struct thread that we can put on lists for testing.
56  */
57 static
58 struct thread *
59 fakethread_create(const char *name)
60 {
61  struct thread *t;
62 
63  t = kmalloc(sizeof(*t));
64  if (t == NULL) {
65  panic("threadlisttest: Out of memory\n");
66  }
67  /* ignore most of the fields, zero everything for tidiness */
68  bzero(t, sizeof(*t));
69  t->t_name = kstrdup(name);
70  if (t->t_name == NULL) {
71  panic("threadlisttest: Out of memory\n");
72  }
73  t->t_stack = FAKE_MAGIC;
74  threadlistnode_init(&t->t_listnode, t);
75  return t;
76 }

94 static
95 void
96 check_order(struct threadlist *tl, bool rev)
97 {
98  const char string0[] = "...";
99  const char stringN[] = "~~~";
100 
101  struct thread *t;
102  const char *first = rev ? stringN : string0;
103  const char *last = rev ? string0 : stringN;
104  const char *prev;
105  int cmp;
106 
107  prev = first;
108  THREADLIST_FORALL(t, *tl) {
109  cmp = strcmp(prev, t->t_name);
110  KASSERT(rev ? (cmp > 0) : (cmp < 0));
111  prev = t->t_name;
112  }
113  cmp = strcmp(prev, last);
114  KASSERT(rev ? (cmp > 0) : (cmp < 0));
115 }
