30 /* matmult.c
31  * Test program to do matrix multiplication on large arrays.
32  *
33  * This version uses a storage-inefficient technique to get a
34  * shorter running time for the same memory usage.
35  *
36  * Intended to stress virtual memory system.
37  *
38  * Once the VM system assignment is complete your system should be
39  * able to survive this.
40  */

47  */

51 int A[Dim][Dim];

56 int
57 main(void)

59  int i, j, k, r;
60 
61  for (i = 0; i < Dim; i++) /* first initialize the matrices */
62  for (j = 0; j < Dim; j++) {
63  A[i][j] = i;
64  B[i][j] = j;
65  C[i][j] = 0;
66  }
67 
68  for (i = 0; i < Dim; i++) /* then multiply them together */
69  for (j = 0; j < Dim; j++)
70  for (k = 0; k < Dim; k++)
71  T[i][j][k] = A[i][k] * B[k][j];
72 
73  for (i = 0; i < Dim; i++)
74  for (j = 0; j < Dim; j++)
75  for (k = 0; k < Dim; k++)
76  C[i][j] += T[i][j][k];
77 
78  r = 0;
79  for (i = 0; i < Dim; i++)
80  r += C[i][i];
81 
82  printf("matmult finished.\n");
83  printf("answer is: %d (should be %d)\n", r, RIGHT);
84  if (r != RIGHT) {
85  printf("FAILED\n");
86  return 1;
87  }
88  printf("Passed.\n");
89  return 0;
