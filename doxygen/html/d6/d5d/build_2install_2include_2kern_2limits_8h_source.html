30 #ifndef _KERN_LIMITS_H_
31 #define _KERN_LIMITS_H_
32 
33 /*
34  * Constants for libc's <limits.h> - system limits.
35  *
36  * The symbols are prefixed with __ here to avoid namespace pollution
37  * in libc. Use <limits.h> (in either userspace or the kernel) to get
38  * the proper names.
39  *
40  * These are Unix-style limits that Unix defines; you can change them
41  * around or add others as needed or as are appropriate to your system
42  * design.
43  *
44  * Likewise, the default values provided here are fairly reasonable,
45  * but you can change them around pretty freely and userspace code
46  * should adapt. Do change these as needed to match your
47  * implementation.
48  */
49 
50 
51 /*
52  * Important, both as part of the system call API and for system behavior.
53  *
54  * 255 for NAME_MAX and 1024 for PATH_MAX are conventional. ARG_MAX
55  * should be at least 16K. In real systems it often runs to 256K or
56  * more.
57  */
58 
59 /* Longest filename (without directory) not including null terminator */
60 #define __NAME_MAX 255
61 
62 /* Longest full path name */
63 #define __PATH_MAX 1024
64 
65 /* Max bytes for an exec function (should be at least 16K) */
66 #define __ARG_MAX (64 * 1024)
67 
68 
69 /*
70  * Important for system behavior, but not a big part of the API.
71  *
72  * Most modern systems don't have OPEN_MAX at all, and instead go by
73  * whatever limit is set with setrlimit().
74  */
75 
76 /* Min value for a process ID (that can be assigned to a user process) */
77 #define __PID_MIN 2
78 
79 /* Max value for a process ID (change this to match your implementation) */
80 #define __PID_MAX 32767
81 
82 /* Max open files per process */
83 #define __OPEN_MAX 128
84 
85 /* Max bytes for atomic pipe I/O -- see description in the pipe() man page */
86 #define __PIPE_BUF 512
87 
88 
89 /*
90  * Not so important parts of the API. (Especially in OS/161 where we
91  * don't do credentials by default.)
92  */
93 
94 /* Max number of supplemental group IDs in process credentials */
95 #define __NGROUPS_MAX 32
96 
97 /* Max login name size (for setlogin/getlogin), incl. null */
98 #define __LOGIN_NAME_MAX 17
99 
100 
101 /*
102  * Not very important at all.
103  */
104 
105 /* Max number of iovec structures at once for readv/writev/preadv/pwritev */
106 #define __IOV_MAX 1024
